20      public void testBasicFilter() {

22          testRules.add(new HttpsSecureCookieRule("", "foo")); // exactly foo
23          testRules.add(new HttpsSecureCookieRule("", "bar.*")); // starts with bar
24          testRules.add(new HttpsSecureCookieRule("", ".*qup")); // ends with qup
25          testRules.add(new HttpsSecureCookieRule("", ".*baz.*")); // contains baz
26          
27          CookieFilter f = new HttpsSecureCookieFilter(testRules);
28          
29          assertFalse(f.accepts(new DefaultCookie("foo", "0")));
30          assertTrue(f.accepts(new DefaultCookie("foox", "0")));
31          assertTrue(f.accepts(new DefaultCookie("xfoo", "0")));
32          assertTrue(f.accepts(new DefaultCookie("xfoox", "0")));
33          
34          assertFalse(f.accepts(new DefaultCookie("bar", "0")));
35          assertFalse(f.accepts(new DefaultCookie("barx", "0")));
36          assertTrue(f.accepts(new DefaultCookie("xbar", "0")));
37          assertTrue(f.accepts(new DefaultCookie("xbarx", "0")));
38
