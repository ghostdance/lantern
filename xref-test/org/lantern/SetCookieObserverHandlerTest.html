108     @Test
109     public void testMalformedSetCookie() {
110 
111         final String requestUri = "http://www.example.org/foo";
112 
113         final SetCookieObserverHandlerTest tester = this;
114         class ObserverTest implements SetCookieObserver {
115 
116             private final Logger log = LoggerFactory.getLogger(getClass());            
117             public boolean called = false; 
118             
119             @Override
120             public void setCookies(Collection<Cookie> cookies, HttpRequest context) {
121                 called = true; 
122                 log.error("Called with cookies: {}", cookies);
123             }
124         }
125         
126         final ObserverTest observer = new ObserverTest();
127         final ChannelHandler handler = new SetCookieObserverHandler(observer);
128 
129         final ChannelPipeline pipeline = Channels.pipeline();
130         final Channel chan = createDummyChannel(pipeline); 
131         pipeline.addLast("set_cookie_observer", handler);
132 
133         // we have to send an httprequest down the pipeline first
134         // to correlate with our response...
135         final HttpRequest req = createGetRequest(requestUri);
136 
137         final String invalid[] = {" ", ";=;,", "+ +"};
138         for (final String val : invalid) {
139             // create an HttpResponse with several Set-Cookies in it...
140             final HttpResponse res = createResponse();
141             res.setHeader(HttpHeaders.Names.SET_COOKIE, val);
142 
143             assertTrue(res.getHeader(HttpHeaders.Names.SET_COOKIE).equals(val));
144 
145             // send the fake request...
146             pipeline.sendUpstream(createDummyMessageEvent(req));
147 
148             // then the fake reply...
149             pipeline.sendDownstream(createDummyMessageEvent(res));
150             
151             // header should be unchanged and observer should be uncalled
152             assertTrue(res.getHeader(HttpHeaders.Names.SET_COOKIE).equals(val));
153             assertFalse(observer.called);
154         }
155
