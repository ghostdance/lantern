22      /**
23       * Sending registry values with quotes to the java external process system
24       * requires extra escaping of quotes, so just make sure we handle it and
25       * get the same results as call to the advapi stuff.
26       * 
27       * @throws Exception If any unexpected error occurs.
28       */
29      @Test
30      public void testRegistryValuesWithQuotes() throws Exception {
31          if (!SystemUtils.IS_OS_WINDOWS) {
32              return;
33          }
34          final String key = "Software\\Microsoft\\Windows\\CurrentVersion\\Run";
35          final String fewerQuotes = 
36              "\""+new File("Lantern.exe").getCanonicalPath()+"\"" + " --launchd";
37          final String name = "Lantern";
38          
39          writeWithAdvApi(key, name, fewerQuotes);
40          String advResult = Registry.read(key, name);
41          writeWithCommandReg(key, name, fewerQuotes);
42          String regResult = Registry.read(key, name);
43          assertEquals(advResult, regResult);
44      }
45  
46      private boolean writeWithAdvApi(final String key, final String name, 
47          final String value) {
48          Advapi32Util.registrySetStringValue(WinReg.HKEY_CURRENT_USER, key, 
49              name, value);
50          return true;
51      }
52      
53      private static boolean writeWithCommandReg(final String key, 
54          final String name, final String value) {
55          final int exit = 
56              WindowsRegCommand.writeREG_SZ("HKCU\\"+key, name, value.toString());
57          final boolean succeeded = exit == 0;
58          if (!succeeded) {
59              LOG.warn("Could not write to reg with REG command either: "+key);
60          } else {
61              LOG.info("Successfully wrote ot registry with REG command");
62          }
63          return succeeded;
64      }
