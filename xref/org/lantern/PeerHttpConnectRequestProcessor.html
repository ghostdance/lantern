19   * HTTP request processor that sends requests to peers.

21  public class PeerHttpConnectRequestProcessor implements HttpRequestProcessor {
22  
23      private final Logger log = LoggerFactory.getLogger(getClass());
24      
25      private final AtomicBoolean configured = new AtomicBoolean(false);

33      private final LanternSocketsUtil socketsUtil;

35      public PeerHttpConnectRequestProcessor(final Socket sock,
36          final ChannelGroup channelGroup, final ByteTracker byteTracker,
37          final LanternSocketsUtil socketsUtil) {
38          this.sock = sock;
39          this.channelGroup = channelGroup;
40          this.byteTracker = byteTracker;
41          this.socketsUtil = socketsUtil;
42      }
43  
44      @Override
45      public boolean processRequest(final Channel browserToProxyChannel,
46          final ChannelHandlerContext ctx, final MessageEvent me) 
47          {
48          
49          if (!configured.getAndSet(true)) {
50              browserToProxyChannel.setReadable(false);
51              // We tell the socket to record stats here because traffic
52              // returning to the browser is just shuttled through 
53              // a SocketHttpConnectRelayingHandler and the normal 
54              // encoder that records stats is removed from the 
55              // browserToProxyChannel pipeline.
56              this.socketsUtil.startReading(this.sock, browserToProxyChannel, true);
57              
58              log.info("Got an outbound socket on request handler hash {} to {}", 
59                  hashCode(), this.sock);
60                  
61              final ChannelPipeline browserPipeline = 
62                  browserToProxyChannel.getPipeline();
63              browserPipeline.remove("encoder");
64              browserPipeline.remove("decoder");
65              browserPipeline.remove("handler");
66              
67              browserPipeline.addLast("handler", 
68                  new SocketHttpConnectRelayingHandler(this.sock, 
69                      this.channelGroup));
70              browserToProxyChannel.setReadable(true);
71          }
72  
73          log.info("Processing request...");
74          // Lantern's a transparent proxy here, so we forward the HTTP CONNECT
75          // message to the remote peer.
76          final HttpRequest request = (HttpRequest) me.getMessage();
77          try {
78              final OutputStream os = this.sock.getOutputStream();
79              final byte[] data = LanternUtils.toByteBuffer(request, ctx);
80              //log.info("Writing data on peer socket: {}", new String(data, "UTF-8"));
81              log.debug("Writing {} bytes on peer socket...", data.length);
82              os.write(data);
83              
84              // Remember this could be any kind of underlying socket here, 
85              // including a UDP socket with an OutputStream that might not
86              // have truly written then bytes even though it's theoretically
87              // blocking.
88              byteTracker.addUpBytes(data.length);
89          } catch (final IOException e) {
90              log.error("Could not write to stream?", e);
91              return false;
92          } catch (final Exception e) {
93              log.error("Could not encode request?", e);
94              return false;
95          }
96          return true;
97      }
