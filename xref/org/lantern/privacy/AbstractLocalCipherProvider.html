138     @Override
139     public synchronized Cipher newLocalCipher(int opmode) throws IOException, 
140         GeneralSecurityException {
141         final boolean init = !isInitialized();
142         
143         if (localKey == null) {
144             log.info("Retrieving local cipher key...");
145             localKey = getLocalKey(init);
146         }
147 
148         final Cipher cipher = getCipher();
149 
150         if (init) {
151             initializeCipher(cipher, opmode, localKey);
152             saveParameters(cipher);
153         }
154         else {
155            final AlgorithmParameters params = loadParameters();
156            cipher.init(opmode, localKey, params);
157         }
158         return cipher;
159     }

186     byte[] createValidator(byte [] key, byte[] salt) throws GeneralSecurityException {
187         // the validator is the concatenation of the salt value 
188         // and the sha256 digest of the salt and the key
189         final MessageDigest md = MessageDigest.getInstance("SHA-256");
190         md.update(salt);
191         md.update(key);
192         final byte[] hash = md.digest();
193         final byte[] validator = new byte[salt.length + hash.length];
194         System.arraycopy(salt, 0, validator, 0, salt.length);
195         System.arraycopy(hash, 0, validator, salt.length, hash.length);
196         return validator; 
197     }
