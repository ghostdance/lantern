32      public PeerCounter(long bucketSizeMillis, long ageLimit, final Timer timer) {
33          super(bucketSizeMillis, ageLimit);
34          Events.register(this);
35          peers = new ConcurrentSkipListSet<String>();
36          lifetimePeers = new ConcurrentSkipListSet<String>();
37          
38          // measure the current number of peers at regular intervals...
39          timer.scheduleAtFixedRate(new TimerTask() {
40              @Override
41              public void run() {
42                  measurePeers();
43              }
44          }, 0, bucketSizeMillis);
45      }
46      
47      @Override
48      public void reset() {
49          super.reset();
50          peers.clear();
51          lifetimePeers.clear();
52      }
53      
54      protected void measurePeers() {
55          long now = System.currentTimeMillis();
56          long bucket = bucketForTimestamp(now);
57  
58          if (lastBucket != -1) {
59              if (bucket == lastBucket) {
60                  log.warn("Peer counter updated faster than normal...");
61                  return;
62              }
63              if (bucket - lastBucket < 0) {
64                  log.warn("...sdrawkcab gninnur si emiT");
65                  return;
66              }
67              if (bucket - lastBucket > 1) {
68                  log.warn("Peer counter thread is running more than a bucket slow...");
69              }
70          }
71          addData(now, peers.size());
72          resetLifetimeTotal(lifetimePeers.size());
73          lastBucket = bucket;
74      }
75      
76      
77      @Subscribe
78      protected void onP2PConnectionEvent(final P2PConnectionEvent event) {
79          // could obscure with hash, but not stored now.
80          final String peerId = event.getJid(); 
81          
82          if (event.isConnected()) {
83              peers.add(peerId);
84              lifetimePeers.add(peerId);
85          }
86          else {
87              peers.remove(peerId);
88          }
89      }
