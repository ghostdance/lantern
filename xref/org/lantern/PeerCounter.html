28      public PeerCounter(long bucketSizeMillis, long ageLimit) {
29          super(bucketSizeMillis, ageLimit);
30          LanternHub.register(this);
31          peers = new ConcurrentSkipListSet<String>();
32          lifetimePeers = new ConcurrentSkipListSet<String>();
33          
34          // measure the current number of peers at regular intervals...
35          final Timer timer = LanternHub.timer();
36          timer.scheduleAtFixedRate(new TimerTask() {
37              @Override
38              public void run() {
39                  measurePeers();
40              }
41          }, 0, bucketSizeMillis);
42      }
43      
44      public void reset() {
45          super.reset();
46          peers.clear();
47          lifetimePeers.clear();
48      }
49      
50      protected void measurePeers() {
51          long now = System.currentTimeMillis();
52          long bucket = bucketForTimestamp(now);
53  
54          if (lastBucket != -1) {
55              if (bucket == lastBucket) {
56                  log.warn("Peer counter updated faster than normal...");
57                  return;
58              }
59              if (bucket - lastBucket < 0) {
60                  log.warn("...sdrawkcab gninnur si emiT");
61                  return;
62              }
63              if (bucket - lastBucket > 1) {
64                  log.warn("Peer counter thread is running more than a bucket slow...");
65              }
66          }
67          addData(now, peers.size());
68          resetLifetimeTotal(lifetimePeers.size());
69          lastBucket = bucket;
70      }
71      
72      
73      @Subscribe
74      protected void onP2PConnectionEvent(final P2PConnectionEvent event) {
75          // could obscure with hash, but not stored now.
76          final String peerId = event.getJid(); 
77          
78          if (event.isConnected()) {
79              peers.add(peerId);
80              lifetimePeers.add(peerId);
81          }
82          else {
83              peers.remove(peerId);
84          }
85      }
