36      /**
37       * Makes the dialog visible.
38       * 
39       * @return The password text.
40       * @throws UserInputRequiredException If the user did not enter any input.
41       */
42      public String askForPassword() throws UserInputRequiredException {
43          final Shell parent = getParent();
44          final Shell shell = new Shell(parent, SWT.TITLE | SWT.BORDER
45                  | SWT.APPLICATION_MODAL);
46          shell.setText("Lantern Password");
47  
48          final RowLayout layout = new RowLayout(SWT.VERTICAL);
49          layout.marginLeft = 20;
50          layout.marginTop = 20;
51          layout.marginRight = 20;
52          layout.marginBottom = 20;
53          layout.center = true;
54          
55          shell.setLayout(layout);
56          
57          if (StringUtils.isNotBlank(errorMessage)) {
58              final Composite errorComposite = newMarginComposite(shell, 2);
59              final Label error = new Label(errorComposite, SWT.CENTER);
60              error.setText(this.errorMessage);
61              error.setForeground(parent.getDisplay().getSystemColor(SWT.COLOR_RED));
62          }
63          final Composite labelComposite = newMarginComposite(shell, 10);
64          final Composite passwordComposite = newMarginComposite(shell, 10);
65          final Composite buttonComposite = newMarginComposite(shell, 10);
66  
67          final Label description = new Label(labelComposite, SWT.CENTER);
68          description.setText("Please enter your Lantern password.");
69          
70          final Text passwordField = 
71              new Text(passwordComposite, SWT.SINGLE | SWT.BORDER | SWT.PASSWORD | SWT.CENTER);
72          
73          passwordField.setLayoutData(new RowData(200, 22));
74          passwordField.setFocus();
75          
76          final Button buttonCancel = new Button(buttonComposite, SWT.PUSH);
77          buttonCancel.setText("Cancel");
78          
79          final Button buttonOK = new Button(buttonComposite, SWT.PUSH);
80          buttonOK.setText("OK");
81          buttonOK.setEnabled(false);
82  
83          final AtomicReference<String> passwordText = 
84              new AtomicReference<String>();
85          
86          passwordField.addListener(SWT.Modify, new Listener() {
87              @Override
88              public void handleEvent(final Event event) {
89                   final String text = passwordField.getText();
90                   if (StringUtils.isNotEmpty(text)) {
91                       buttonOK.setEnabled(true);
92                   } else {
93                       buttonOK.setEnabled(false);
94                   }
95              }
96          });
97  
98          buttonOK.addListener(SWT.Selection, new Listener() {
99              @Override
100             public void handleEvent(final Event event) {
101                 passwordText.set(passwordField.getText());
102                 shell.dispose();
103             }
104         });
105 
106         buttonCancel.addListener(SWT.Selection, new Listener() {
107             @Override
108             public void handleEvent(final Event event) {
109                 shell.dispose();
110             }
111         });
112         
113         shell.addListener(SWT.Traverse, new Listener() {
114             @Override
115             public void handleEvent(final Event event) {
116                 if (event.detail == SWT.TRAVERSE_ESCAPE) {
117                     event.doit = false;
118                 }
119             }
120         });
121 
122         shell.pack();
123         final Rectangle rect = shell.getBounds();
124         final Point center = 
125             LanternUtils.getScreenCenter(rect.width, rect.height);
126         shell.setLocation((int)center.getX(), (int)center.getY());
127         
128         passwordField.setText("");
129         
130         shell.pack();
131         shell.open();
132         shell.forceActive();
133 
134         final Display display = parent.getDisplay();
135         while (!shell.isDisposed()) {
136             if (!display.readAndDispatch()) {
137                 display.sleep();
138             }
139         }
140 
141         final String text = passwordText.get();
142         if (StringUtils.isEmpty(text)) {
143             throw new UserInputRequiredException();
144         }
145         //shell.close();
146         //shell.getDisplay().dispose();
147         shell.dispose();
148         display.dispose();
149         return text;
150     }

152    private Composite newMarginComposite(final Shell shell, final int marginBottom) {
153        final Composite comp = new Composite(shell, SWT.NONE);
154        final RowLayout layout = new RowLayout();
155        layout.marginBottom = marginBottom;
156        layout.center = true;
157        layout.pack = true;
158        layout.type = SWT.HORIZONTAL;
159        comp.setLayout(layout);
160        return comp;
161     }
