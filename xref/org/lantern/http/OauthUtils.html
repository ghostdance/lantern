34      public static GoogleClientSecrets loadClientSecrets() throws IOException {
35          if (secrets != null) {
36              return secrets;
37          }
38          InputStream is = null;
39          try {
40              is = OauthUtils.class.getResourceAsStream(
41                  "/client_secrets_installed.json");
42              secrets = GoogleClientSecrets.load(new JacksonFactory(), is);
43              //log.debug("Secrets: {}", secrets);
44              return secrets;
45          } finally {
46              IOUtils.closeQuietly(is);
47          }
48      }

50      /**
51       * Utility class for making an Oauth request to a Google service.
52       * 
53       * @param access The access token.
54       * @param refresh The refresh token.
55       * @param encodedUrl The URL to visit.
56       * 
57       * @return The {@link HttpResponse}.
58       * @throws IOException If there's an error loading the client secrets or
59       * accessing the service.
60       */
61      public static HttpResponse googleOauth(final String access, 
62          final String refresh, final String encodedUrl) throws IOException{
63          
64          final GoogleClientSecrets creds = OauthUtils.loadClientSecrets();
65          final CredentialRefreshListener refreshListener = 
66              new CredentialRefreshListener() {
67                  
68                  @Override
69                  public void onTokenResponse(final Credential credential,
70                      final TokenResponse tokenResponse) throws IOException {
71                      LOG.info("Got token response...sending event");
72                      Events.eventBus().post(new TokenResponseEvent(tokenResponse));
73                  }
74                  
75                  @Override
76                  public void onTokenErrorResponse(final Credential credential,
77                      final TokenErrorResponse tokenErrorResponse) 
78                      throws IOException {
79                      LOG.warn("Error response:\n"+
80                              tokenErrorResponse.toPrettyString());
81                  }
82              };
83          final GoogleCredential gc = new GoogleCredential.Builder().
84              setTransport(new NetHttpTransport()).
85              setJsonFactory(new JacksonFactory()).
86              addRefreshListener(refreshListener).
87              setClientAuthentication(new ClientParametersAuthentication(
88                  creds.getInstalled().getClientId(), 
89                  creds.getInstalled().getClientSecret())).build();
90          
91          gc.setAccessToken(access);
92          gc.setRefreshToken(refresh);
93  
94          final GenericUrl url = new GenericUrl(encodedUrl);
95          final HttpRequestFactory requestFactory = 
96              gc.getTransport().createRequestFactory(gc);
97          return requestFactory.buildGetRequest(url).execute();
98          
99      }
