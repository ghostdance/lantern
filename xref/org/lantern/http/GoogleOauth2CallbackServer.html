21  /**
22   * This is a special server that runs on the port we have registered for
23   * OAuth callbacks with Google. It should be started every time we know we
24   * need OAuth (every time the OAuth redirect page is hit), and should stop it
25   * as soon as the callback is done.
26   */
27  public class GoogleOauth2CallbackServer {
28  
29      private final Logger log = LoggerFactory.getLogger(getClass());
30      
31      private final Server server = new Server();
32      
33      private final int port = 7777;
34  
35      private final XmppHandler xmppHandler;
36  
37      private final Model model;
38  
39      private final InternalState internalState;
40  
41      private final ModelIo modelIo;
42  
43      private final Proxifier proxifier;
44      
45      public GoogleOauth2CallbackServer(final XmppHandler xmppHandler,
46          final Model model, final InternalState internalState,
47          final ModelIo modelIo, final Proxifier proxifier) {
48          this.xmppHandler = xmppHandler;
49          this.model = model;
50          this.internalState = internalState;
51          this.modelIo = modelIo;
52          this.proxifier = proxifier;
53      }
54      
55      public void start() {
56          // Note we unfortunately can't give our threads names using this
57          // thread pool.
58          server.setThreadPool(new ExecutorThreadPool(1));
59          final String apiName = "Lantern-Oauth-Callback";
60          final ContextHandlerCollection contexts = 
61              new ContextHandlerCollection();
62          
63          final ServletContextHandler contextHandler = newContext("/", apiName);
64          contexts.addHandler(contextHandler);
65  
66          
67          server.setHandler(contexts);
68          server.setStopAtShutdown(true);
69          
70          final SelectChannelConnector connector = 
71              new SelectChannelConnector();
72          log.info("Setting connector port for OAUTH to: {}", port);
73          connector.setPort(port);
74          connector.setMaxIdleTime(80 * 1000);
75          connector.setLowResourcesMaxIdleTime(30 * 1000);
76          connector.setLowResourcesConnections(100);
77          connector.setHost("127.0.0.1");
78          connector.setName(apiName);
79          connector.setAcceptors(1);
80          
81          this.server.setConnectors(new Connector[]{connector});
82          
83          final ServletHolder oauth2callback = new ServletHolder(
84              new GoogleOauth2CallbackServlet(this, this.xmppHandler, 
85                  this.model, this.internalState, this.modelIo, this.proxifier));
86          oauth2callback.setInitOrder(1);
87          contextHandler.addServlet(oauth2callback, "/oauth2callback");
88          
89          final Thread serve = new Thread(new Runnable() {
90  
91              @Override
92              public void run() {
93                  try {
94                      server.start();
95                      server.join();
96                  } catch (final Exception e) {
97                      log.error("Exception on HTTP server");
98                  }
99              }
100             
101         }, "HTTP-Server-Oauth-Thread");
102         serve.setDaemon(true);
103         serve.start();
104         
105         log.info("About to wait for server....");
106         LanternUtils.waitForServer(this.port);
107     }

109     private ServletContextHandler newContext(final String path,
110         final String name) {
111         final ServletContextHandler context = 
112             new ServletContextHandler(ServletContextHandler.SESSIONS);
113         
114         final Map<String, String> params = context.getInitParams();
115         params.put("org.eclipse.jetty.servlet.Default.gzip", "false");
116         params.put("org.eclipse.jetty.servlet.Default.welcomeServlets", "false");
117         params.put("org.eclipse.jetty.servlet.Default.dirAllowed", "false");
118         params.put("org.eclipse.jetty.servlet.Default.aliases", "false");
119         context.setContextPath(path);
120         context.setConnectorNames(new String[] {name});
121         return context;
122     }
