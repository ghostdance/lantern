18  /**
19   * NAT-PMP service class that wraps the underlying NAT-PMP implementation
20   * from flszen.
21   */
22  public class NatPmp implements NatPmpService {
23  
24      private final Logger log = LoggerFactory.getLogger(getClass());
25      
26      private NatPmpDevice pmpDevice;
27      
28      private final List<MapRequestMessage> requests =
29          new ArrayList<MapRequestMessage>();
30      
31      /**
32       * Creates a new NAT-PMP instance.
33       * 
34       * @throws NatPmpException If we could not start NAT-PMP for any reason.
35       */
36      public NatPmp() throws NatPmpException {
37          if (NetworkUtils.isPublicAddress()) {
38              // If we're already on the public network, there's no NAT.
39              return;
40          }
41          pmpDevice = new NatPmpDevice(false);
42          
43          // We implement the shutdown hook ourselves so we can explicitly 
44          // remove all the mappings we've created.
45          Runtime.getRuntime().addShutdownHook(new Thread(new Runnable() {
46  
47              @Override
48              public void run() {
49                  // Remove all the mappings and shutdown.
50                  log.info("Shutting down NAT-PMP");
51                  final int num = requests.size();
52                  for (int i = 0; i < num; i++) {
53                      removeNatPmpMapping(i);
54                  }
55                  //pmpDevice.shutdown();
56                  log.info("Finished shutdown for NAT-PMP");
57              }
58              
59          }, "NAT-PMP-Shutdown-Thread"));
60      }
61      
62      @Override
63      public void removeNatPmpMapping(final int mappingIndex) {
64          log.info("Removing mapping...");
65          if (NetworkUtils.isPublicAddress()) {
66              return;
67          }
68          final MapRequestMessage request = requests.get(mappingIndex);
69          
70          final boolean tcp = request.getMessageType() == MessageType.MapTCP;
71          
72          // Setting the lifetime to zero removes the mapping.
73          final MapRequestMessage remove = 
74              new MapRequestMessage(tcp, request.getInternalPort(), 
75                  request.getRequestedExternalPort(), 0, null);
76          pmpDevice.enqueueMessage(remove);
77          log.info("About to wait on queue");
78          pmpDevice.waitUntilQueueEmpty();
79          log.info("Finished waiting on queue");
80      }
81      
82      @Override
83      public int addNatPmpMapping(final PortMappingProtocol prot, 
84          final int localPort, final int externalPortRequested,
85          final PortMapListener portMapListener) {
86          if (NetworkUtils.isPublicAddress()) {
87              // If we're already on the public network, there's no NAT.
88              return 1;
89          }
90          if (portMapListener == null) {
91              log.error("No listener?");
92              throw new NullPointerException("Null listener");
93          }
94          // This call will block unless we thread it here.
95          final Runnable natPmpRunner = new Runnable() {
96              @Override
97              public void run() {
98                  // Note we don't pass the requested external port -- with
99                  // NAT-PMP we just use whatever the router gives us.
100                 addMapping(prot, localPort, portMapListener);
101             }
102         };
103         final Thread mapper = new Thread(natPmpRunner, "NAT-PMP-Mapping-Thread");
104         mapper.setDaemon(true);
105         final int index = requests.size();
106         mapper.start();
107         
108         return index;
109     }
110 
111     protected void addMapping(final PortMappingProtocol prot,
112         final int localPort, final PortMapListener portMapListener) {
113         log.info("Adding NAT-PMP mapping");
114         final boolean tcp;
115         if (prot == PortMappingProtocol.TCP) {
116             tcp = true;
117         } else {
118             tcp = false;
119         }
120         // We just take whatever port the router gives us, ignoring the 
121         // requested port.
122         final int lifeTimeSeconds = 60 * 60;
123         final MapRequestMessage map = 
124             new MapRequestMessage(tcp, localPort, 0, lifeTimeSeconds, null);
125         pmpDevice.enqueueMessage(map);
126         //pmpDevice.waitUntilQueueEmpty();
127         try {
128             // Auto-boxing can cause a null pointer here, so make sure to
129             // use Integer.
130             final Integer extPort = map.getExternalPort();
131             if (extPort != null) { 
132                 log.info("Got external port!! "+extPort);
133                 portMapListener.onPortMap(extPort);
134                 LanternHub.statsTracker().setNatpmp(true);
135             } else {
136                 portMapListener.onPortMapError();
137                 LanternHub.statsTracker().setNatpmp(false);
138             }
139         } catch (final NatPmpException e) {
140             portMapListener.onPortMapError();
141             LanternHub.statsTracker().setNatpmp(false);
142         }
143         // We have to add it whether it succeeded or not to keep the indeces 
144         // in sync.
145         requests.add(map);
146     }
