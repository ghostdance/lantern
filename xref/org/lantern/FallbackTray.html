17   * A SystemTray implementation that falls back among available alternatives.

19  @Singleton //@Named("facade")
20  public class FallbackTray implements SystemTray {
21      private final Logger log = LoggerFactory.getLogger(getClass());
22      private SystemTray nonLinuxTray;
23      private AppIndicatorTray linuxTray;
24      private final Model model;
25      
26  
27      @Inject
28      public FallbackTray(final SystemTray nonLinuxTray,
29          final AppIndicatorTray linuxTray, final Model model) {
30          this.nonLinuxTray = nonLinuxTray;
31          this.linuxTray = linuxTray;
32          this.model = model;
33          this.linuxTray.setFailureCallback(new AppIndicatorTray.FailureCallback() {
34              @Override
35              public void createTrayFailed() {
36                  fallback();
37              }
38          });
39          Events.register(this);
40      }
41      
42  
43      @Override
44      public void start() {
45          createTray();
46      }
47  
48      @Override
49      public void stop() {
50          
51      }
52      
53      @Override
54      public void createTray() {
55          if (SystemUtils.IS_OS_LINUX && model.getSettings().isUiEnabled() && 
56              this.linuxTray.isSupported()) {
57              this.linuxTray.createTray(); // may call fallback() later...
58          }
59          else {
60              fallback(); // fall back immediately
61          }
62      }
63  
64      @Subscribe
65      public void onUpdate(final UpdateEvent update) {
66          if (nonLinuxTray != null) {
67              nonLinuxTray.addUpdate(update.getData());
68          }
69      }
70      
71      @Override
72      public void addUpdate(final Map<String, Object> updateData) {
73          nonLinuxTray.addUpdate(updateData);
74      }
75  
76      @Override
77      public boolean isActive() {
78          return nonLinuxTray != null && nonLinuxTray.isActive();
79      }
80   
81      public void fallback() {
82          log.debug("App Indicator tray is not available.");
83          if (model.getSettings().isUiEnabled() && nonLinuxTray.isSupported()) {
84              log.debug("Falling back to SWT Tray.");
85              nonLinuxTray.createTray();
86          }
87          else {
88              log.info("Disabling tray.");
89              nonLinuxTray = new SystemTray() {
90                  @Override
91                  public void createTray() {}
92                  @Override
93                  public void addUpdate(Map<String, Object> updateData) {}
94                  @Override
95                  public boolean isActive() {return false;}
96                  @Override
97                  public boolean isSupported() {return false;}
98                  @Override
99                  public void start() throws Exception {}
100                 @Override
101                 public void stop() {}
102             };
103         }
104     }
