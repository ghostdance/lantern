68      public SSLServerSocketFactory newTlsServerSocketFactory() {
69          log.info("Creating TLS server socket factory");
70          try {
71              final KeyManagerFactory kmf = loadKeyManagerFactory(getSslAlgorithm());
72              
73              // Initialize the SSLContext to work with our key managers.
74              final SSLContext serverContext = SSLContext.getInstance("TLS");
75              serverContext.init(kmf.getKeyManagers(), ksm.getTrustManagers(), null);

77          } catch (final NoSuchAlgorithmException e) {
78              throw new Error("Could not create SSL server socket factory.", e);

93      private KeyManagerFactory loadKeyManagerFactory(final String algorithm) {
94          try {
95              final KeyStore ks = KeyStore.getInstance("JKS");
96              ks.load(ksm.keyStoreAsInputStream(), ksm.getKeyStorePassword());
97              
98              // Set up key manager factory to use our key store
99              final KeyManagerFactory kmf = KeyManagerFactory.getInstance(algorithm);
100             kmf.init(ks, ksm.getCertificatePassword());
101             return kmf;
102         } catch (final KeyStoreException e) {
103             throw new Error("Key manager issue", e);
104         } catch (final UnrecoverableKeyException e) {
105             throw new Error("Key manager issue", e);
106         } catch (final NoSuchAlgorithmException e) {
107             throw new Error("Key manager issue", e);
108         } catch (final CertificateException e) {
109             throw new Error("Key manager issue", e);
110         } catch (final IOException e) {
111             throw new Error("Key manager issue", e);
112         }
113     }

262     public void startReading(final Socket sock, final Channel channel, 
263         final boolean recordStats) {
264         final Runnable runner = new Runnable() {
265             @Override
266             public void run() {
267                 final byte[] buffer = new byte[4096];
268                 int n = 0;
269                 try {
270                     log.info("READING FROM SOCKET: {}", sock);
271                     if (sock.isClosed()) {
272                         log.error("SOCKET IS CLOSED");
273                         ProxyUtils.closeOnFlush(channel);
274                         return;
275                     }
276                     final InputStream is = sock.getInputStream();
277                     while (-1 != (n = is.read(buffer))) {
278                         //log.info("Writing response data: {}", new String(buffer, 0, n));
279                         // We need to make a copy of the buffer here because
280                         // the writes are asynchronous, so the bytes can
281                         // otherwise get scrambled.
282                         final ChannelBuffer buf =
283                             ChannelBuffers.copiedBuffer(buffer, 0, n);
284                         channel.write(buf);
285                         if (recordStats) {
286                             stats.addDownBytesFromPeers(n);
287                         }
288                         
289                     }
290                     ProxyUtils.closeOnFlush(channel);
291 
292                 } catch (final IOException e) {
293                     log.info("Exception relaying peer data back to browser", e);
294                     ProxyUtils.closeOnFlush(channel);
295                     
296                     // The other side probably just closed the connection!!
297                     
298                     //channel.close();
299                     //proxyStatusListener.onError(peerUri);
300                     
301                 }
302             }
303         };
304         threadPool.execute(runner);
305     }
