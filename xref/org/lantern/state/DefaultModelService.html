72      @Override
73      public void setRunAtSystemStart(final boolean runOnSystemStartup) {
74          log.debug("Setting start at login to "+runOnSystemStartup);
75          
76          this.model.getSettings().setRunAtSystemStart(runOnSystemStartup);
77          Events.sync(SyncPath.START_AT_LOGIN, runOnSystemStartup);
78          if (SystemUtils.IS_OS_MAC_OSX && this.launchdPlist.isFile()) {
79              setStartAtLoginOsx(runOnSystemStartup);
80          } else if (SystemUtils.IS_OS_WINDOWS) {
81              setStartAtLoginWindows(runOnSystemStartup);
82          } else if (SystemUtils.IS_OS_LINUX) {
83              log.info("Setting setStartAtLogin for Linux");
84              setStartAtLoginLinux(runOnSystemStartup);
85          } else {
86              log.warn("setStartAtLogin not yet implemented for {}", SystemUtils.OS_NAME);
87          }
88      }

100     public void setStartAtLoginWindows(final boolean start) {
101         final String key = 
102             "Software\\Microsoft\\Windows\\CurrentVersion\\Run";
103         int result = 0;
104         if (start) {
105             try {
106                 final String path = 
107                     "\""+new File("Lantern.exe").getCanonicalPath()+"\"" + " --launchd";
108                     //"\"\\\""+new File("Lantern.exe").getCanonicalPath()+"\\\"\"" + " --launchd";
109                 
110                 
111                 Registry.write(key, "Lantern", path);
112             } catch (final IOException e) {
113                 log.error("Could not get canonical path", e);
114             }
115         } else {
116             Registry.write(key, "Lantern", "");
117         }

119         if (result != 0) {
120             log.error("Error changing startAtLogin? Result: "+result);
121         }

135     @Override
136     public void setSystemProxy(final boolean isSystemProxy) {
137         if (isSystemProxy == this.model.getSettings().isSystemProxy()) {
138             log.info("System proxy setting is unchanged.");
139             return;
140         }
141         this.model.getSettings().setSystemProxy(isSystemProxy);
142         
143         log.info("Setting system proxy");
144         
145         // Go ahead and change the setting so that it will affect. It will 
146         // be set again by the api, but that doesn't matter.
147         this.model.getSettings().setSystemProxy(isSystemProxy);
148         if (!this.model.isSetupComplete()) {
149             return;
150         }
151         
152         final Runnable proxyRunner = new Runnable() {
153             @Override
154             public void run() {
155                 try {
156                     if (modelUtils.shouldProxy() ) {
157                         proxifier.startProxying();
158                     } else {
159                         proxifier.stopProxying();
160                     }
161                 } catch (final Proxifier.ProxyConfigurationError e) {
162                     log.error("Proxy reconfiguration failed: {}", e);
163                 }
164             }
165         };
166         proxyQueue.execute(proxyRunner);
167     }

169     @Override
170     public void setMode(final Mode mode) {
171         log.debug("Calling set get mode. Get is: "+mode);
172         // When we move to give mode, we want to start advertising our 
173         // ID and to start accepting incoming connections.
174         
175         // We we move to get mode, we want to stop advertising our ID and to
176         // stop accepting incoming connections.
177         final Settings set = this.model.getSettings();
178         if (mode == set.getMode()) {
179             log.info("Mode is unchanged.");
180             return;
181         }
182         
183         // Go ahead and set the setting although it will also be
184         // updated by the api as well. We want to make sure the
185         // state seen by the following calls is consistent with
186         // this flag being aspirational vs. representational
187         set.setMode(mode);
188         
189         // We disconnect and reconnect to create a new Jabber ID that will 
190         // not advertise us as a connection point.
191         if (!model.isSetupComplete()) {
192             log.debug("Not disconnecting and reconnecting before setup is " +
193                 "complete");
194             return;
195         }
196         
197         // We dont' want to force the frontend to wait for all of this, so we
198         // thread it.
199         final Runnable runner = new Runnable() {
200 
201             @Override
202             public void run() {
203                 xmppHandler.disconnect();
204                 try {
205                     try {
206                         xmppHandler.connect();
207                         
208                         // TODO: This isn't quite right. We don't necessarily have
209                         // proxies to connect to at this point, and we shouldn't set
210                         // the OS proxy until we do.
211                         // may need to modify the proxying state
212                         if (modelUtils.shouldProxy()) {
213                             proxifier.startProxying();
214                         } else {
215                             proxifier.stopProxying();
216                         }
217                     } catch (final IOException e) {
218                         log.info("Could not connect to server", e);
219                         // Don't proxy if there's some error connecting.
220                         proxifier.stopProxying();
221                     } catch (final CredentialException e) {
222                         log.info("Credentials are wrong!!");
223                         proxifier.stopProxying();
224                     } catch (final NotInClosedBetaException e) {
225                         log.info("Not in beta!!");
226                         proxifier.stopProxying();
227                     }
228                 } catch (final Proxifier.ProxyConfigurationError e) {
229                     log.info("Proxy auto-configuration failed: {}", e);
230                 }
231             }
232         };
233         
234         final Thread t = new Thread(runner, "Mode-Shift-Thread");
235         t.setDaemon(true);
236         t.start();
237     }
