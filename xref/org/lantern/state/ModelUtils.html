35      /**
36       * This is used for when the user disconnects and reconnects for any reason.
37       * We store the users we know to have been in the closed beta so we don't
38       * need to wait for a response from the lantern XMPP bot if we already
39       * know they're invited.
40       * 
41       * Note the user will typically only be connecting with one IP addres, but
42       * this setup ensures that any architecture changes the may change that 
43       * won't affect invite lookups.
44       * 
45       * @param email The email to check.
46       * @return <code>true</code> if we already know the specified user to be in
47       * the closed beta. This will return <code>false</code> if we just don't
48       * know -- they could be in or the could not be, but we haven't verified
49       * they are in.
50       */
51      public boolean isInClosedBeta(final String email) {
52          final Set<String> in = this.model.getSettings().getInClosedBeta();
53          return in.contains(email);
54      }

67      public boolean isConfigured() {
68          if (!LanternConstants.DEFAULT_MODEL_FILE.isFile()) {
69              LOG.debug("No settings file");
70              return false;
71          }
72          final String refresh = this.model.getSettings().getRefreshToken();
73          final boolean oauth = this.model.getSettings().isUseGoogleOAuth2();
74          final boolean hasRefresh = StringUtils.isNotBlank(refresh);
75          
76          LOG.debug("Has refresh: "+hasRefresh);
77          LOG.debug("Has oauth: "+oauth);
78          return oauth && hasRefresh;
79      }

98      public void loadOAuth2ClientSecretsFile(final String filename) {
99          if (StringUtils.isBlank(filename)) {
100             LOG.error("No filename specified");
101             throw new NullPointerException("No filename specified!");
102         }
103         final File file = new File(filename);
104         if (!(file.exists() && file.canRead())) {
105             LOG.error("Unable to read user credentials from {}", filename);
106             throw new IllegalArgumentException("File does not exist! "+filename);
107         }
108         LOG.info("Reading client secrets from file \"{}\"", filename);
109         try {
110             final String json = FileUtils.readFileToString(file, "US-ASCII");
111             JSONObject obj = (JSONObject)JSONValue.parse(json);
112             final JSONObject ins;
113             final JSONObject temp = (JSONObject)obj.get("installed");
114             if (temp == null) {
115                 ins = (JSONObject)obj.get("web");
116             } else {
117                 ins = temp;
118             }
119             //JSONObject ins = (JSONObject)obj.get("installed");
120             final String clientID = (String)ins.get("client_id");
121             final String clientSecret = (String)ins.get("client_secret");
122             if (StringUtils.isBlank(clientID) || 
123                 StringUtils.isBlank(clientSecret)) {
124                 LOG.error("Failed to parse client secrets file \"{}\"", file);
125                 throw new Error("Failed to parse client secrets file: "+ file);
126             } else {
127                 this.model.getSettings().setClientID(clientID);
128                 this.model.getSettings().setClientSecret(clientSecret);
129             }
130         } catch (final IOException e) {
131             LOG.error("Failed to read file \"{}\"", filename);
132             throw new Error("Could not load oauth file"+file, e);
133         }
134     }

136     public void loadOAuth2UserCredentialsFile(final String filename) {
137         if (StringUtils.isBlank(filename)) {
138             LOG.error("No filename specified");
139             throw new NullPointerException("No filename specified!");
140         }
141         final File file = new File(filename);
142         if (!(file.exists() && file.canRead())) {
143             LOG.error("Unable to read user credentials from {}", filename);
144             throw new IllegalArgumentException("File does not exist! "+filename);
145         }
146         LOG.info("Reading user credentials from file \"{}\"", filename);
147         try {
148             final String json = FileUtils.readFileToString(file, "US-ASCII");
149             final JSONObject obj = (JSONObject)JSONValue.parse(json);
150             final String username = (String)obj.get("username");
151             final String accessToken = (String)obj.get("access_token");
152             final String refreshToken = (String)obj.get("refresh_token");
153             // Access token is not strictly necessary, so we allow it to be
154             // null.
155             if (StringUtils.isBlank(username) || 
156                 StringUtils.isBlank(refreshToken)) {
157                 LOG.error("Failed to parse user credentials file \"{}\"", filename);
158                 throw new Error("Could not load username or refresh_token");
159             } else {
160                 this.model.getSettings().setEmail(username);
161                 //this.model.getSettings().setCommandLineEmail(username);
162                 this.model.getSettings().setAccessToken(accessToken);
163                 this.model.getSettings().setRefreshToken(refreshToken);
164                 this.model.getSettings().setUseGoogleOAuth2(true);
165             }
166         } catch (final IOException e) {
167             LOG.error("Failed to read file \"{}\"", filename);
168             throw new Error("Could not load oauth credentials", e);
169         }
170     }
