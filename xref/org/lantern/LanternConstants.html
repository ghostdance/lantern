197             }
198             if (!LOG_DIR.isDirectory()) {

200                     System.err.println("Could not create dir at: " + LOG_DIR);
201                 }
202             }

205                     System.err.println("Could not make config directory at: "+
206                         CONFIG_DIR);
207                 }
208             }
