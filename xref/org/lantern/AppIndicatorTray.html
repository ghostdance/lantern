90      public void createTray() {
91  
92          /*uniqueApp = libunique.unique_app_new("org.lantern.lantern", null);
93          if (libunique.unique_app_is_running(uniqueApp)) {
94              LOG.error("Already running!");
95              System.exit(0);

98          
99          menu = libgtk.gtk_menu_new();
100         
101         connectionStatusItem = libgtk.gtk_menu_item_new_with_label(LABEL_DISCONNECTED);
102         libgtk.gtk_widget_set_sensitive(connectionStatusItem, Gtk.FALSE);
103         libgtk.gtk_menu_shell_append(menu, connectionStatusItem);
104         libgtk.gtk_widget_show_all(connectionStatusItem);
105         
106         dashboardItem = libgtk.gtk_menu_item_new_with_label("Open Dashboard");
107         dashboardItemCallback = new Gobject.GCallback() {
108             @Override
109             public void callback(Pointer instance, Pointer data) {
110                 LOG.debug("openDashboardItem callback called");
111                 openDashboard();
112             }
113         };
114         libgobject.g_signal_connect_data(dashboardItem, "activate", dashboardItemCallback,null, null, 0);
115         libgtk.gtk_menu_shell_append(menu, dashboardItem);
116         libgtk.gtk_widget_show_all(dashboardItem);
117         
118         //updateItem = Gtk.gtk_menu_item_new_with_label();
119         
120         quitItem = libgtk.gtk_menu_item_new_with_label("Quit");
121         quitItemCallback = new Gobject.GCallback() {
122             @Override
123             public void callback(Pointer instance, Pointer data) {
124                 LOG.debug("quitItemCallback called");
125                 quit();
126             }
127         };
128         libgobject.g_signal_connect_data(quitItem, "activate", quitItemCallback,null, null, 0);
129         libgtk.gtk_menu_shell_append(menu, quitItem);
130         libgtk.gtk_widget_show_all(quitItem);
131         
132         appIndicator = libappindicator.app_indicator_new(
133             "lantern", "indicator-messages-new",
134             AppIndicator.CATEGORY_APPLICATION_STATUS);
135         
136         /* XXX basically a hack -- we should subclass the AppIndicator 
137            type and override the fallback entry in the 'vtable', instead we just
138            hack the app indicator class itself. Not an issue unless we need other 
139            appindicators. 
140         */
141         AppIndicator.AppIndicatorClassStruct aiclass = 
142             new AppIndicator.AppIndicatorClassStruct(appIndicator.parent.g_type_instance.g_class);
143         
144         AppIndicator.Fallback replacementFallback = new AppIndicator.Fallback() {
145             @Override
146             public Pointer callback(AppIndicator.AppIndicatorInstanceStruct self) {
147                 fallback();
148                 return null;
149             }
150         };
151 
152         aiclass.fallback = replacementFallback;
153         aiclass.write();
154 
155         libappindicator.app_indicator_set_menu(appIndicator, menu);
156         
157         changeIcon(ICON_DISCONNECTED, LABEL_DISCONNECTED);
158         libappindicator.app_indicator_set_status(appIndicator, AppIndicator.STATUS_ACTIVE);
159     
160         LanternHub.register(this);
161     }

193     @Override
194     public void addUpdate(final Map<String, Object> data) { 
195         LOG.info("Adding update data: {}", data);
196         if (this.updateData != null && this.updateData.equals(data)) {
197             LOG.info("Ignoring duplicate update data");
198             return;
199         }
200         this.updateData = data;
201         final String label = I18n.tr("Update to Lantern ") + 
202             data.get(LanternConstants.UPDATE_VERSION_KEY);
203         updateItem = libgtk.gtk_menu_item_new_with_label(label);
204         updateItemCallback = new Gobject.GCallback() {
205             @Override
206             public void callback(Pointer instance, Pointer pointer) {
207                 LOG.debug("updateItemCallback called");
208                 NativeUtils.openUri(
209                     (String)data.get(LanternConstants.UPDATE_URL_KEY));
210             }
211         };
212         libgobject.g_signal_connect_data(updateItem, "activate", updateItemCallback,null, null, 0);
213         libgtk.gtk_menu_shell_append(menu, updateItem);
214         libgtk.gtk_widget_show_all(updateItem);
215     }

222     @Subscribe
223     public void onConnectivityStateChanged(
224         final ConnectivityStatusChangeEvent csce) {
225         final ConnectivityStatus cs = csce.getConnectivityStatus();
226         LOG.info("Got connectivity state changed {}", cs);
227         switch (cs) {
228         case DISCONNECTED: {
229             changeIcon(ICON_DISCONNECTED, LABEL_DISCONNECTED);
230             break;
231         }
232         case CONNECTING: {
233             changeIcon(ICON_CONNECTING, LABEL_CONNECTING);
234             break;
235         }
236         case CONNECTED: {
237             changeIcon(ICON_CONNECTED, LABEL_CONNECTED);
238             break;
239         }
240         }
241     }
