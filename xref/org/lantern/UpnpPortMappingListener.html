43  /**
44   * Maintains UPnP port mappings on an InternetGatewayDevice automatically.
45   * <p>
46   * This listener will wait for discovered devices which support either
47   * {@code WANIPConnection} or the {@code WANPPPConnection} service. As soon as any such
48   * service is discovered, the desired port mapping will be created. When the UPnP service
49   * is shutting down, all previously established port mappings with all services will
50   * be deleted.
51   * </p>
52   * <p>
53   * The following listener maps external WAN TCP port 8123 to internal host 10.0.0.2:
54   * </p>
55   * <pre>{@code
56   * upnpService.getRegistry().addListener(
57   *newPortMappingListener(newPortMapping(8123, "10.0.0.2",PortMapping.Protocol.TCP))
58   * );}</pre>
59   * <p>
60   * If all you need from the Cling UPnP stack is NAT port mapping, use the following idiom:
61   * </p>
62   * <pre>{@code
63   * UpnpService upnpService = new UpnpServiceImpl(
64   *     new PortMappingListener(new PortMapping(8123, "10.0.0.2", PortMapping.Protocol.TCP))
65   * );
66   * <p/>
67   * upnpService.getControlPoint().search(new STAllHeader()); // Search for all devices
68   * <p/>
69   * upnpService.shutdown(); // When you no longer need the port mapping
70   * }</pre>
71   *
72   * @author Christian Bauer
73   * Modified for use in Lantern by Adam Fisk
74   */
75  public class UpnpPortMappingListener extends DefaultRegistryListener {
76  
77      private static final Logger log = 
78          LoggerFactory.getLogger(UpnpPortMappingListener.class);
79  
80      public static final DeviceType IGD_DEVICE_TYPE = new UDADeviceType("InternetGatewayDevice", 1);
81      public static final DeviceType CONNECTION_DEVICE_TYPE = new UDADeviceType("WANConnectionDevice", 1);
82  
83      public static final ServiceType IP_SERVICE_TYPE = new UDAServiceType("WANIPConnection", 1);
84      public static final ServiceType PPP_SERVICE_TYPE = new UDAServiceType("WANPPPConnection", 1);
85  
86      protected PortMapping[] portMappings;
87  
88      // The key of the map is Service and equality is object identity, this is by-design
89      protected Map<Service, List<PortMapping>> activePortMappings = new HashMap();
90  
91      private final PortMapListener portMapListener;
92  
93      private final Stats stats;
94  
95      public UpnpPortMappingListener(final Stats stats, 
96          final PortMapListener portMapListener, 
97          final PortMapping... portMappings) {
98          this.stats = stats;
99          this.portMapListener = portMapListener;
100         this.portMappings = portMappings;
101     }
102 
103     @Override
104     synchronized public void deviceAdded(Registry registry, Device device) {
105 
106         final Service connectionService;
107         if ((connectionService = discoverConnectionService(device)) == null) return;
108 
109         log.debug("Activating port mappings on: " + connectionService);
110 
111         final List<PortMapping> activeForService = new ArrayList();
112         final Action action = connectionService.getAction("AddPortMapping");
113         if (action == null) {
114             log.info("No ACTION in UPnP mapping attempt!");
115             return;
116         }
117         for (final PortMapping pm : portMappings) {
118             new PortMappingAdd(connectionService, registry.getUpnpService().getControlPoint(), pm) {
119 
120                 @Override
121                 public void success(ActionInvocation invocation) {
122                     log.debug("Port mapping added: " + pm);
123                     stats.setUpnp(true);
124                     activeForService.add(pm);
125                     portMapListener.onPortMap((int) pm.getExternalPort().getValue().longValue());
126                 }
127 
128                 @Override
129                 public void failure(ActionInvocation invocation, UpnpResponse operation, String defaultMsg) {
130                     handleFailureMessage("Failed to add port mapping: " + pm);
131                     handleFailureMessage("Reason: " + defaultMsg);
132                     stats.setUpnp(false);
133                     portMapListener.onPortMapError();
134                 }
135             }.run(); // Synchronous!
136         }
137 
138         activePortMappings.put(connectionService, activeForService);
139     }
140 
141     @Override
142     synchronized public void deviceRemoved(Registry registry, Device device) {
143         for (Service service : device.findServices()) {
144             Iterator<Map.Entry<Service, List<PortMapping>>> it = activePortMappings.entrySet().iterator();
145             while (it.hasNext()) {
146                 Map.Entry<Service, List<PortMapping>> activeEntry = it.next();
147                 if (!activeEntry.getKey().equals(service)) continue;
148 
149                 if (activeEntry.getValue().size() > 0)
150                     handleFailureMessage("Device disappeared, couldn't delete port mappings: " + activeEntry.getValue().size());
151 
152                 it.remove();
153             }
154         }
155     }
156 
157     @Override
158     synchronized public void beforeShutdown(Registry registry) {
159         for (Map.Entry<Service, List<PortMapping>> activeEntry : activePortMappings.entrySet()) {
160 
161             final Iterator<PortMapping> it = activeEntry.getValue().iterator();
162             while (it.hasNext()) {
163                 final PortMapping pm = it.next();
164                 log.debug("Trying to delete port mapping on IGD: " + pm);
165                 new PortMappingDelete(activeEntry.getKey(), registry.getUpnpService().getControlPoint(), pm) {
166 
167                     @Override
168                     public void success(ActionInvocation invocation) {
169                         log.debug("Port mapping deleted: " + pm);
170                         it.remove();
171                     }
172 
173                     @Override
174                     public void failure(ActionInvocation invocation, UpnpResponse operation, String defaultMsg) {
175                         handleFailureMessage("Failed to delete port mapping: " + pm);
176                         handleFailureMessage("Reason: " + defaultMsg);
177                     }
178 
179                 }.run(); // Synchronous!
180             }
181         }
182     }
183 
184     protected Service discoverConnectionService(Device device) {
185         if (!device.getType().equals(IGD_DEVICE_TYPE)) {
186             return null;
187         }
188 
189         Device[] connectionDevices = device.findDevices(CONNECTION_DEVICE_TYPE);
190         if (connectionDevices.length == 0) {
191             log.debug("IGD doesn't support '" + CONNECTION_DEVICE_TYPE + "': " + device);
192             return null;
193         }
194 
195         final Device connectionDevice = connectionDevices[0];
196         log.debug("Using first discovered WAN connection device: " + connectionDevice);
197 
198         final Service ipConnectionService = 
199             connectionDevice.findService(IP_SERVICE_TYPE);
200         final Service pppConnectionService = 
201             connectionDevice.findService(PPP_SERVICE_TYPE);
202 
203         if (ipConnectionService == null && pppConnectionService == null) {
204             log.debug("IGD doesn't support IP or PPP WAN connection service: " + device);
205         }
206 
207         return ipConnectionService != null ? ipConnectionService : pppConnectionService;
208     }
209 
210     protected void handleFailureMessage(final String s) {
211         log.warn(s);
212     }
