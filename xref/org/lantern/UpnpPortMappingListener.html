43  /**
44   * Maintains UPnP port mappings on an InternetGatewayDevice automatically.
45   * <p>
46   * This listener will wait for discovered devices which support either
47   * {@code WANIPConnection} or the {@code WANPPPConnection} service. As soon as any such
48   * service is discovered, the desired port mapping will be created. When the UPnP service
49   * is shutting down, all previously established port mappings with all services will
50   * be deleted.
51   * </p>
52   * <p>
53   * The following listener maps external WAN TCP port 8123 to internal host 10.0.0.2:
54   * </p>
55   * <pre>{@code
56   * upnpService.getRegistry().addListener(
57   *newPortMappingListener(newPortMapping(8123, "10.0.0.2",PortMapping.Protocol.TCP))
58   * );}</pre>
59   * <p>
60   * If all you need from the Cling UPnP stack is NAT port mapping, use the following idiom:
61   * </p>
62   * <pre>{@code
63   * UpnpService upnpService = new UpnpServiceImpl(
64   *     new PortMappingListener(new PortMapping(8123, "10.0.0.2", PortMapping.Protocol.TCP))
65   * );
66   * <p/>
67   * upnpService.getControlPoint().search(new STAllHeader()); // Search for all devices
68   * <p/>
69   * upnpService.shutdown(); // When you no longer need the port mapping
70   * }</pre>
71   *
72   * @author Christian Bauer
73   * Modified for use in Lantern by Adam Fisk
74   */
75  public class UpnpPortMappingListener extends DefaultRegistryListener {
76  
77      private static final Logger log = 
78          LoggerFactory.getLogger(UpnpPortMappingListener.class);
79  
80      public static final DeviceType IGD_DEVICE_TYPE = new UDADeviceType("InternetGatewayDevice", 1);
81      public static final DeviceType CONNECTION_DEVICE_TYPE = new UDADeviceType("WANConnectionDevice", 1);
82  
83      public static final ServiceType IP_SERVICE_TYPE = new UDAServiceType("WANIPConnection", 1);
84      public static final ServiceType PPP_SERVICE_TYPE = new UDAServiceType("WANPPPConnection", 1);
85  
86      protected PortMapping[] portMappings;
87  
88      // The key of the map is Service and equality is object identity, this is by-design
89      protected Map<Service, List<PortMapping>> activePortMappings = new HashMap();
90  
91      private final PortMapListener portMapListener;
92  
93      public UpnpPortMappingListener(final PortMapListener portMapListener, 
94          final PortMapping... portMappings) {
95          this.portMapListener = portMapListener;
96          this.portMappings = portMappings;
97      }
98  
99      @Override
100     synchronized public void deviceAdded(Registry registry, Device device) {
101 
102         final Service connectionService;
103         if ((connectionService = discoverConnectionService(device)) == null) return;
104 
105         log.debug("Activating port mappings on: " + connectionService);
106 
107         final List<PortMapping> activeForService = new ArrayList();
108         final Action action = connectionService.getAction("AddPortMapping");
109         if (action == null) {
110             log.info("No ACTION in UPnP mapping attempt!");
111             return;
112         }
113         for (final PortMapping pm : portMappings) {
114             new PortMappingAdd(connectionService, registry.getUpnpService().getControlPoint(), pm) {
115 
116                 @Override
117                 public void success(ActionInvocation invocation) {
118                     log.debug("Port mapping added: " + pm);
119                     activeForService.add(pm);
120                     portMapListener.onPortMap((int) pm.getExternalPort().getValue().longValue());
121                 }
122 
123                 @Override
124                 public void failure(ActionInvocation invocation, UpnpResponse operation, String defaultMsg) {
125                     handleFailureMessage("Failed to add port mapping: " + pm);
126                     handleFailureMessage("Reason: " + defaultMsg);
127                     portMapListener.onPortMapError();
128                 }
129             }.run(); // Synchronous!
130         }
131 
132         activePortMappings.put(connectionService, activeForService);
133     }
134 
135     @Override
136     synchronized public void deviceRemoved(Registry registry, Device device) {
137         for (Service service : device.findServices()) {
138             Iterator<Map.Entry<Service, List<PortMapping>>> it = activePortMappings.entrySet().iterator();
139             while (it.hasNext()) {
140                 Map.Entry<Service, List<PortMapping>> activeEntry = it.next();
141                 if (!activeEntry.getKey().equals(service)) continue;
142 
143                 if (activeEntry.getValue().size() > 0)
144                     handleFailureMessage("Device disappeared, couldn't delete port mappings: " + activeEntry.getValue().size());
145 
146                 it.remove();
147             }
148         }
149     }
150 
151     @Override
152     synchronized public void beforeShutdown(Registry registry) {
153         for (Map.Entry<Service, List<PortMapping>> activeEntry : activePortMappings.entrySet()) {
154 
155             final Iterator<PortMapping> it = activeEntry.getValue().iterator();
156             while (it.hasNext()) {
157                 final PortMapping pm = it.next();
158                 log.debug("Trying to delete port mapping on IGD: " + pm);
159                 new PortMappingDelete(activeEntry.getKey(), registry.getUpnpService().getControlPoint(), pm) {
160 
161                     @Override
162                     public void success(ActionInvocation invocation) {
163                         log.debug("Port mapping deleted: " + pm);
164                         it.remove();
165                     }
166 
167                     @Override
168                     public void failure(ActionInvocation invocation, UpnpResponse operation, String defaultMsg) {
169                         handleFailureMessage("Failed to delete port mapping: " + pm);
170                         handleFailureMessage("Reason: " + defaultMsg);
171                     }
172 
173                 }.run(); // Synchronous!
174             }
175         }
176     }
177 
178     protected Service discoverConnectionService(Device device) {
179         if (!device.getType().equals(IGD_DEVICE_TYPE)) {
180             return null;
181         }
182 
183         Device[] connectionDevices = device.findDevices(CONNECTION_DEVICE_TYPE);
184         if (connectionDevices.length == 0) {
185             log.debug("IGD doesn't support '" + CONNECTION_DEVICE_TYPE + "': " + device);
186             return null;
187         }
188 
189         final Device connectionDevice = connectionDevices[0];
190         log.debug("Using first discovered WAN connection device: " + connectionDevice);
191 
192         final Service ipConnectionService = 
193             connectionDevice.findService(IP_SERVICE_TYPE);
194         final Service pppConnectionService = 
195             connectionDevice.findService(PPP_SERVICE_TYPE);
196 
197         if (ipConnectionService == null && pppConnectionService == null) {
198             log.debug("IGD doesn't support IP or PPP WAN connection service: " + device);
199         }
200 
201         return ipConnectionService != null ? ipConnectionService : pppConnectionService;
202     }
203 
204     protected void handleFailureMessage(final String s) {
205         log.warn(s);
206     }
207 
208 }
