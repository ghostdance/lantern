110     @Provides @Singleton
111     LookupService provideLookupService() {
112         final File unzipped = 
113                 new File(LanternConstants.DATA_DIR, "GeoIP.dat");
114         if (!unzipped.isFile())  {
115             final File file = new File("GeoIP.dat.gz");
116             GZIPInputStream is = null;
117             OutputStream os = null;
118             try {
119                 is = new GZIPInputStream(new FileInputStream(file));
120                 os = new FileOutputStream(unzipped);
121                 IOUtils.copy(is, os);
122             } catch (final IOException e) {
123                 log.error("Error expanding file?", e);
124             } finally {
125                 IOUtils.closeQuietly(is);
126                 IOUtils.closeQuietly(os);
127             }
128         }
129         try {
130             return new LookupService(unzipped, 
131                     LookupService.GEOIP_MEMORY_CACHE);
132         } catch (final IOException e) {
133             log.error("Could not create LOOKUP service?");
134         }
135         return null;
136     }

220     /**
221      * Copies our FireFox extension to the appropriate place.
222      * 
223      * @return The {@link File} for the final destination directory of the
224      * extension.
225      * @throws IOException If there's an error copying the extension.
226      */
227     public void copyFireFoxExtension() throws IOException {
228         log.info("Copying FireFox extension");
229         final File dir = getExtensionDir();
230         if (!dir.isDirectory()) {
231             log.info("Making FireFox extension directory...");
232             // NOTE: This likely means the user does not have FireFox. We copy
233             // the extension here anyway in case the user ever installs 
234             // FireFox in the future.
235             if (!dir.mkdirs()) {
236                 log.error("Could not create ext dir: "+dir);
237                 throw new IOException("Could not create ext dir: "+dir);
238             }
239         }
240         final String extName = "lantern@getlantern.org";
241         final File dest = new File(dir, extName);
242         final File ffDir = new File("firefox/"+extName);
243         if (dest.exists() && !FileUtils.isFileNewer(ffDir, dest)) {
244             log.info("Extension already exists and ours is not newer");
245             return;
246         }
247         if (!ffDir.isDirectory()) {
248             log.error("No extension directory found at {}", ffDir);
249             throw new IOException("Could not find extension?");
250         }
251         FileUtils.copyDirectoryToDirectory(ffDir, dir);
252         log.info("Copied FireFox extension from {} to {}", ffDir, dir);
253     }

257         if (SystemUtils.IS_OS_WINDOWS) {

259             return new File(ffDir, "Extensions/{ec8030f7-c20a-464f-9b0e-13a3a9e97384}");
260         } else if (SystemUtils.IS_OS_MAC_OSX) {
261             return new File(userHome, 
262                 "Library/Application Support/Mozilla/Extensions/{ec8030f7-c20a-464f-9b0e-13a3a9e97384}");
263         } else {
264             return new File(userHome, "Mozilla/extensions/{ec8030f7-c20a-464f-9b0e-13a3a9e97384}");
265         }
