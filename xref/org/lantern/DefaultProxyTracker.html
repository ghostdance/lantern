91      @Override
92      public boolean addPeerProxy(final URI peerUri) {
93          log.info("Considering peer proxy");
94          synchronized (peerProxySet) {
95              // We purely do this to keep track of which peers we've attempted
96              // to establish connections to. This is to avoid exchanging certs
97              // multiple times.
98              
99              // TODO: I believe this excludes exchanging keys with peers who
100             // are on multiple machines when the peer URI is a general JID and
101             // not an instance JID.
102             if (!peerProxySet.contains(peerUri)) {
103                 log.info("Actually adding peer proxy: {}", peerUri);
104                 peerProxySet.add(peerUri);
105                 return true;
106             } else {
107                 log.info("We already know about the peer proxy");
108             }
109         }
110         return false;
111     }

121     @Override
122     public void addGeneralProxy(final String cur) {
123         final String hostname = StringUtils.substringBefore(cur, ":");
124         final int port = Integer.parseInt(StringUtils.substringAfter(cur, ":"));
125         final InetSocketAddress isa = new InetSocketAddress(hostname, port);
126         addProxyWithChecks(proxySet, proxies, new ProxyHolder(hostname, isa), 
127             cur);

130     private InetSocketAddress getProxy(final Queue<ProxyHolder> queue) {
131         synchronized (queue) {
132             if (queue.isEmpty()) {
133                 log.debug("No proxy addresses");
134                 return null;
135             }
136             final ProxyHolder proxy = queue.remove();
137             queue.add(proxy);
138             log.debug("FIFO queue is now: {}", queue);
139             return proxy.isa;
140         }
141     }

143     private void addProxyWithChecks(final Set<ProxyHolder> set,
144         final Queue<ProxyHolder> queue, final ProxyHolder ph, 
145         final String fullProxyString) {
146         if (set.contains(ph)) {
147             log.debug("We already know about proxy "+ph+" in {}", set);
148             return;
149         }
150         
151         final Socket sock = new Socket();
152         try {
153             sock.connect(ph.isa, 60*1000);
154             log.debug("Dispatching CONNECTED event");
155             
156             // This is a little odd because the proxy could have originally
157             // come from the settings themselves, but it'll remove duplicates,
158             // so no harm done.
159             log.debug("Adding proxy to settings: {}", this.model.getSettings());
160             this.model.getSettings().addProxy(fullProxyString);
161             synchronized (set) {
162                 if (!set.contains(ph)) {
163                     set.add(ph);
164                     queue.add(ph);
165                     log.debug("Queue is now: {}", queue);
166                 }
167             }
168         } catch (final IOException e) {
169             log.error("Could not connect to: {}", ph);
170             onCouldNotConnect(ph.isa);
171             this.model.getSettings().removeProxy(fullProxyString);
172         } finally {
173             IOUtils.closeQuietly(sock);
174         }
175     }

202     private void onCouldNotConnect(final ProxyHolder proxyAddress,
203         final Set<ProxyHolder> set, final Queue<ProxyHolder> queue){
204         log.info("COULD NOT CONNECT!! Proxy address: {}", proxyAddress);
205         synchronized (this.proxySet) {
206             set.remove(proxyAddress);
207             queue.remove(proxyAddress);

265     private static final class ProxyHolder {

267         private final String id;
268         private final InetSocketAddress isa;

270         private ProxyHolder(final String id, final InetSocketAddress isa) {
271             this.id = id;
272             this.isa = isa;

275         @Override
276         public String toString() {
277             return "ProxyHolder [isa=" + isa + "]";
278         }
279         
280         @Override
281         public int hashCode() {
282             final int prime = 31;
283             int result = 1;
284             result = prime * result + ((id == null) ? 0 : id.hashCode());
285             result = prime * result + ((isa == null) ? 0 : isa.hashCode());
286             return result;
287         }
288 
289         @Override
290         public boolean equals(Object obj) {
291             if (this == obj)
292                 return true;
293             if (obj == null)
294                 return false;
295             if (getClass() != obj.getClass())
296                 return false;
297             ProxyHolder other = (ProxyHolder) obj;
298             if (id == null) {
299                 if (other.id != null)
300                     return false;
301             } else if (!id.equals(other.id))
302                 return false;
303             if (isa == null) {
304                 if (other.isa != null)
305                     return false;
306             } else if (!isa.equals(other.isa))
307                 return false;
308             return true;
309         }
