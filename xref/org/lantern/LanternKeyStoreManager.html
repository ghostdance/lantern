48      public LanternKeyStoreManager(final File rootDir) {
49          CONFIG_DIR = rootDir != null ? rootDir : LanternConstants.CONFIG_DIR;
50          KEYSTORE_FILE = 
51              new File(CONFIG_DIR, "lantern_keystore.jks");
52          TRUSTSTORE_FILE = 
53              new File(CONFIG_DIR, "lantern_truststore.jks");
54          CERT_FILE = 
55              new File(CONFIG_DIR, "local_lantern_cert");
56          
57          fullDelete(KEYSTORE_FILE);
58          fullDelete(TRUSTSTORE_FILE);
59  
60          if (!CONFIG_DIR.isDirectory()) {
61              if (!CONFIG_DIR.mkdir()) {
62                  log.error("Could not create config dir!! "+CONFIG_DIR);
63              }
64          }
65          reset(LanternUtils.getMacAddress());
66          createTrustStore();
67          
68          this.lanternTrustManager = 
69              new LanternTrustManager(this, TRUSTSTORE_FILE, PASS);
70          
71          trustManagers = new TrustManager[] {
72              lanternTrustManager
73          };
74          Runtime.getRuntime().addShutdownHook(new Thread (new Runnable() {
75              @Override
76              public void run() {
77                  fullDelete(KEYSTORE_FILE);
78                  fullDelete(TRUSTSTORE_FILE);
79              }
80          }, "Keystore-Delete-Thread"));
81      }

90      private void createTrustStore() {
91          if (TRUSTSTORE_FILE.isFile()) {
92              log.info("Trust store already exists");
93              return;
94          }
95          final String result = LanternUtils.runKeytool("-genkey", "-alias", 
96              "foo", "-keysize", KEYSIZE, "-validity", "365", "-keyalg", ALG, 
97              "-dname", "CN="+LanternUtils.getMacAddress(), "-keystore", 
98              TRUSTSTORE_FILE.getAbsolutePath(), "-keypass", PASS, 
99              "-storepass", PASS);
100         log.info("Got result of creating trust store: {}", result);
101     }

103     private void reset(final String macAddress) {
104         log.info("RESETTING KEYSTORE AND TRUSTSTORE!!");
105         if (KEYSTORE_FILE.isFile()) {
106             log.info("Deleting existing keystore file at: " +
107                 KEYSTORE_FILE.getAbsolutePath());
108             KEYSTORE_FILE.delete();
109         }
110         
111         if (TRUSTSTORE_FILE.isFile()) {
112             log.info("Deleting existing truststore file at: " +
113                 TRUSTSTORE_FILE.getAbsolutePath());
114             TRUSTSTORE_FILE.delete();
115         }
116     
117         // Note we use DSA instead of RSA because apparently only the JDK 
118         // has RSA available.
119         final String genKeyResult = LanternUtils.runKeytool("-genkey", "-alias", 
120             macAddress, "-keysize", KEYSIZE, "-validity", "365", "-keyalg", ALG, 
121             "-dname", "CN="+macAddress, "-keypass", PASS, "-storepass", 
122             PASS, "-keystore", KEYSTORE_FILE.getAbsolutePath());
123         
124         
125         log.info("Result of keytool -genkey call: {}", genKeyResult);
126         
127         waitForFile(KEYSTORE_FILE);
128         
129         // Now grab our newly-generated cert. All of our trusted peers will
130         // use this to connect.
131         final String exportCertResult = LanternUtils.runKeytool("-exportcert", "-alias", 
132             macAddress, "-keystore", KEYSTORE_FILE.getAbsolutePath(), 
133             "-storepass", PASS, "-file", CERT_FILE.getAbsolutePath());
134         log.info("Result of keytool -exportcert call: {}", exportCertResult);
135         waitForFile(CERT_FILE);
136         
137         try {
138             final InputStream is = new FileInputStream(CERT_FILE);
139             localCert = Base64.encodeBase64String(IOUtils.toByteArray(is));
140         } catch (final FileNotFoundException e) {
141             log.error("Cert file not found at "+CERT_FILE, e);
142             throw new Error("Cert file not found", e);
143         } catch (final IOException e) {
144             log.error("Could not base 64 encode cert?", e);
145             throw new Error("Could not base 64 encode cert?", e);
146         }
147 
148 
149         /*
150         log.info("Importing cert");
151         nativeCall("keytool", "-import", "-noprompt", "-file", CERT_FILE.getName(), 
152             "-alias", AL, "-keystore", TRUSTSTORE_FILE.getName(), "-storepass", 
153             PASS);
154             */

158     /**
159      * The completion of the native calls is dependent on OS process 
160      * scheduling, so we need to wait until files actually exist.
161      * 
162      * @param file The file to wait for.
163      */
164     private void waitForFile(final File file) {
165         int i = 0;
166         while (!file.isFile() && i < 20) {
167             try {
168                 Thread.sleep(80);
169                 i++;
170             } catch (final InterruptedException e) {
171                 log.error("Interrupted?", e);
172             }
173         }
174     }
