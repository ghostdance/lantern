10  /**
11   * Registry helper class that uses various fallback methods to access the 
12   * registry. This only accesses HKEY_CURRENT_USER because non-admin accounts
13   * cannot read and write to anything else. It's therefore not necessary to
14   * include the root key in any of these calls.
15   */
16  public class Registry {
17      
18      private static final Logger LOG = LoggerFactory.getLogger(Registry.class);
19      
20      public static String read(final String key, final String name) {
21          try {
22              return Advapi32Util.registryGetStringValue(WinReg.HKEY_CURRENT_USER, 
23                  key, name);
24          } catch (final Win32Exception e) {
25              LOG.error("Cannot write to registry using JNA "+key, e);
26              return readWithCommandReg(key, name);
27          }
28      }
29      
30      private static String readWithCommandReg(final String key, 
31          final String name) {
32          return WindowsRegCommand.read("HKCU\\"+key, name);
33      }
34  
35      public static int readInt(final String key, final String name) {
36          try {
37              return Advapi32Util.registryGetIntValue(WinReg.HKEY_CURRENT_USER, 
38                  key, name);
39          } catch (final Win32Exception e) {
40              LOG.error("Cannot write to registry using JNA "+key, e);
41              return Integer.parseInt(readWithCommandReg(key, name));
42          }
43      }
44      
45      public static boolean write(final String key, final String name, 
46          final String value) {
47          try {
48              Advapi32Util.registrySetStringValue(WinReg.HKEY_CURRENT_USER, key, 
49                  name, value);
50              return true;
51          
52          } catch (final Win32Exception e) {
53              LOG.error("Cannot write to registry using JNA "+key, e);
54              return writeWithCommandReg(key, name, value);
55          }
56      }
57  
58      public static boolean write(final String key, final String name, 
59          final Integer value) {
60          try {
61              Advapi32Util.registrySetIntValue(WinReg.HKEY_CURRENT_USER, key, 
62                  name, value);
63              return true;
64          
65          } catch (final Win32Exception e) {
66              LOG.error("Cannot write to registry using JNA "+key, e);
67              return writeWithCommandReg(key, name, value);
68          }
69      }
70      
71      
72      private static boolean writeWithCommandReg(final String key, 
73          final String name, final Integer value) {
74          final int exit = 
75                  WindowsRegCommand.writeREG_DWORD("HKCU\\"+key, name, value);
76              final boolean succeeded = exit == 0;
77              if (!succeeded) {
78                  LOG.warn("Could not write to reg with REG command either: "+key);
79              } else {
80                  LOG.info("Successfully wrote ot registry with REG command");
81              }
82              return succeeded;
83      }

86          final String name, final String value) {
87          final int exit =

89          final boolean succeeded = exit == 0;
