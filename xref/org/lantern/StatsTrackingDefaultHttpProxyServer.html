122     @Override
123     public void start(final boolean localOnly, final boolean anyAddress) {
124         log.info("Starting proxy on port: "+this.port);
125         final HttpServerPipelineFactory factory =
126             new StatsTrackingHttpServerPipelineFactory(authenticationManager,
127                 this.allChannels, this.chainProxyManager, 
128                 new StatsTrackingDefaultRelayPipelineFactoryFactory(chainProxyManager,
129                     this.responseFilters, this.requestFilter,
130                     this.allChannels, this.timer), this.clientChannelFactory, 
131                     this.timer, this.ksm);
132         serverBootstrap.setPipelineFactory(factory);
133 
134         // Binding only to localhost can significantly improve the security of
135         // the proxy.
136         InetSocketAddress isa;
137         if (localOnly) {
138             isa = new InetSocketAddress("127.0.0.1", port);
139         }
140         else if (anyAddress) {
141             isa = new InetSocketAddress(port);
142         } else {
143             try {
144                 isa = new InetSocketAddress(NetworkUtils.getLocalHost(), port);
145             } catch (final UnknownHostException e) {
146                 log.error("Could not get local host?", e);
147                 isa = new InetSocketAddress(port);
148             }
149         }
150         final Channel channel = serverBootstrap.bind(isa);
151         allChannels.add(channel);
152 
153         Runtime.getRuntime().addShutdownHook(new Thread(new Runnable() {
154             @Override
155             public void run() {
156                 stop();
157             }
158         }));
159         log.debug("Started "+this);
160     }

176     private class StatsTrackingHttpServerPipelineFactory 
177         extends HttpServerPipelineFactory {
178         
179         private StatsTrackingHttpServerPipelineFactory(
180             final ProxyAuthorizationManager authorizationManager, 
181             final ChannelGroup channelGroup, 
182             final ChainProxyManager chainProxyManager, 
183             final RelayPipelineFactoryFactory relayPipelineFactoryFactory,
184             final ClientSocketChannelFactory clientChannelFactory, 
185             final Timer timer,
186             final KeyStoreManager ksm) {
187             super(authorizationManager, channelGroup, chainProxyManager, 
188                 ksm, relayPipelineFactoryFactory, 
189                 timer, clientChannelFactory);
190         }
191 
192         @Override
193         public ChannelPipeline getPipeline() throws Exception {
194             final ChannelPipeline pipeline = super.getPipeline();
195             pipeline.addFirst("stats", new StatsTrackingHandler() {
196                 @Override
197                 public void addUpBytes(long bytes) {
198                     stats.addUpBytesToPeers(bytes);
199                 }
200                 @Override
201                 public void addDownBytes(long bytes) {
202                     stats.addDownBytesFromPeers(bytes);
203                 }
204             });
205             return pipeline;
206         }
207     }
