24  /**
25   * HTTP request processor that sends requests to peers and 
26   * writes HttpResponses to the browerToProxy channel. 
27   * 
28   * This differs from PeerHttpRequestProcessor in that it
29   * wraps the littleshoot peer socket in a netty Channel with 
30   * a pipeline that decodes the HttpResponse rather than 
31   * relaying raw bytes. We do this so that we can observe
32   * characteristics of the response in the main browserToProxy
33   * channel pipeline, eg observing Set-Cookie responses.
34   */
35  public class PeerChannelHttpRequestProcessor implements HttpRequestProcessor {
36  
37      private final Logger log = LoggerFactory.getLogger(getClass());
38  
39      private volatile boolean startedCopying;
40      private volatile PeerSocketChannel peerChannel;
41      
42      private final Socket sock;
43      private final PeerSink peerSink = new PeerSink();
44  
45      private final ChannelGroup channelGroup;
46  
47      private final ByteTracker byteTracker;
48  
49      public PeerChannelHttpRequestProcessor(final Socket sock, 
50          final ChannelGroup channelGroup, final ByteTracker byteTracker) {
51          this.sock = sock;
52          this.channelGroup = channelGroup;
53          this.byteTracker = byteTracker;
54      }
55  
56      @Override
57      public boolean processRequest(final Channel browserToProxyChannel,
58          final ChannelHandlerContext ctx, final MessageEvent me) 
59          throws IOException {
60          if (!startedCopying) {
61              final ChannelHandler stats = new StatsTrackingHandler() {
62                  @Override
63                  public void addUpBytes(final long bytes) {
64                      byteTracker.addUpBytes(bytes);
65                  }
66                  @Override
67                  public void addDownBytes(final long bytes) {
68                      byteTracker.addDownBytes(bytes);
69                  }
70              };
71              
72              final ChannelPipeline pipeline = Channels.pipeline();
73              pipeline.addLast("stats", stats);
74              pipeline.addLast("decoder", new HttpResponseDecoder());
75              pipeline.addLast("encoder", new HttpRequestEncoder());
76              pipeline.addLast("relay", 
77                  new RelayToBrowserHandler(browserToProxyChannel));
78  
79              this.peerChannel = new PeerSocketChannel(pipeline, peerSink, sock);
80              this.peerChannel.simulateConnect();
81              startedCopying = true;
82          }
83  
84          final HttpRequest request = (HttpRequest) me.getMessage();
85          Channels.write(peerChannel, request);
86          
87          // We return true in all these case to preserve the behavior before
88          // the change to return a boolean. The point of returning a boolean
89          // was more to consolidate the check for the existence of a proxy with
90          // the request processing.
91          return true;
92      }
93  
94      @Override
95      public boolean processChunk(final ChannelHandlerContext ctx, 
96          final MessageEvent me) throws IOException {
97          final HttpChunk chunk = (HttpChunk) me.getMessage();
98          Channels.write(peerChannel, chunk);
99          return true;
100     }
101 
102     @Override
103     public void close() {
104         ProxyUtils.closeOnFlush(peerChannel);
105     }

107     // This is similar to OutboundHandler, unclear if we need similar complexity
108     // here for range requests, the old version relayed raw bytes, so this 
109     // seems sufficient.
110     private class RelayToBrowserHandler extends SimpleChannelUpstreamHandler {
111         
112         private final Channel browserToProxyChannel;
113         
114         public RelayToBrowserHandler(final Channel browserToProxyChannel) {
115             this.browserToProxyChannel = browserToProxyChannel;
116         }
117 
118         @Override
119         public void messageReceived(ChannelHandlerContext ctx, MessageEvent me) {
120             browserToProxyChannel.write(me.getMessage());
121         }
122         
123         @Override
124         public void channelOpen(final ChannelHandlerContext ctx, 
125             final ChannelStateEvent cse) throws Exception {
126             final Channel ch = cse.getChannel();
127             log.info("New channel opened: {}", ch);
128             channelGroup.add(ch);
129         }
130         
131         @Override
132          public void channelClosed(final ChannelHandlerContext ctx, 
133              final ChannelStateEvent e) throws Exception {
134              log.info("Channel to peer proxy closed, closing browserToProxy channel.");
135              ProxyUtils.closeOnFlush(browserToProxyChannel);
136          }
137 
138          @Override
139          public void exceptionCaught(final ChannelHandlerContext ctx, 
140              final ExceptionEvent e) throws Exception {
141              log.error("Caught exception on peer proxy channel", e.getCause());
142              Channels.close(e.getChannel()); 
143          }
144     }
