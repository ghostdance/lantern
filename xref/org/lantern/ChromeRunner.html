36      private String determineExecutable() throws IOException {
37          final String path = determineExecutablePath();
38          final File file = new File(path);
39          if (!file.isFile()) {
40              throw new IOException("Could not find chrome at:" + path);
41          } else if (!file.canExecute()) {
42              throw new IOException("Chrome not executable at:" + path);
43          }
44          return path;
45      }

47      private String determineExecutablePath() {
48          if (SystemUtils.IS_OS_MAC_OSX) {
49              return "/Applications/Google Chrome.app/Contents/MacOS/Google Chrome";
50          } else if (SystemUtils.IS_OS_LINUX) {
51              final String path1 = "/usr/bin/google-chrome";
52              final File opt1 = new File(path1);
53              if (opt1.isFile() && opt1.canExecute()) return path1;
54              return "/usr/bin/chromium-browser";
55          } else if (SystemUtils.IS_OS_WINDOWS_XP) {
56              final String ad = System.getenv("APPDATA");
57              return ad + "/Google/Chrome/Application/chrome.exe";
58          } else if (SystemUtils.IS_OS_WINDOWS) {
59              return findWindowsExe("APPDATA", "LOCALAPPDATA", "PROGRAMFILES", 
60                  "ProgramW6432");
61          }
62          /*
63           * Should be something like:
64           * 
65           * Windows XP    %HOMEPATH%\Local Settings\Application Data\Google\Chrome\Application\chrome.exe
66           * Windows Vista    C:\Users\%USERNAME%\AppData\Local\Google\Chrome\Application\chrome.exe
67           */
68          throw new UnsupportedOperationException("This is an experimental feature!");
69      }

71      private String findWindowsExe(final String... opts) {
72          final String chromePath = "/Google/Chrome/Application/chrome.exe";
73          for (final String opt : opts) {
74              final String base = System.getenv(opt);
75              final String path = base + chromePath;
76              final File candidate = new File(path);
77              if (candidate.isFile() && candidate.canExecute()) {
78                  return opt;
79              }
80          }
81          throw new UnsupportedOperationException(
82              "Could not find Chrome on Windows!! Searched paths:\n"+
83                      Arrays.asList(opts));
84      }

90      public void open(final int port) throws IOException {
91  
92          if (this.process != null) {
93              try {
94                  final int exitValue = this.process.exitValue();
95                  log.info("Got exit value from former process: ", exitValue);
96              } catch (final IllegalThreadStateException e) {
97                  // This indicates the existing process is still running.
98                  log.info("Ignoring open call since process is still running");
99                  return;
100             }
101         }
102         final String endpoint = StaticSettings.getLocalEndpoint(port);
103         log.debug("Opening browser to: {}", endpoint);
104         final List<String> commands = new ArrayList<String>();
105         final String executable = determineExecutable();
106         commands.add(executable);
107         commands.add("--user-data-dir="+LanternConstants.CONFIG_DIR.getAbsolutePath());
108         commands.add("--window-size="+screenWidth+","+screenHeight);
109         commands.add("--window-position="+location.x+","+location.y);
110         commands.add("--app="+endpoint);
111 
112         final ProcessBuilder processBuilder = new ProcessBuilder(commands);
113         
114         // Note we don't call waitFor on the process to avoid blocking the
115         // calling thread and because we don't care too much about the return
116         // value.
117         this.process = processBuilder.start();
118         
119         new Analyzer(process.getInputStream());
120         new Analyzer(process.getErrorStream());
121     }

123     public void close() {
124         log.info("Closing Chrome browser...process is: {}", this.process);
125         if (this.process != null) {
126             log.info("Really closing Chrome browser...");
127             this.process.destroy();
128             this.process = null;
129         }
130     }

132     private class Analyzer implements Runnable {

134         final InputStream is;

136         public Analyzer(final InputStream is) {
137             this.is = is;
138             final Thread t = new Thread(this);
139             t.setDaemon(true);
140             t.start();
141         }
142 
143         @Override
144         public void run() {
145             BufferedReader br = null;
146             try {
147                 br = new BufferedReader(new InputStreamReader(this.is, 
148                     LanternConstants.UTF8));
149 
150                 String line = "";
151                 while((line = br.readLine()) != null) {
152                     System.out.println(line);
153                 }
154             } catch (final IOException e) {
155                 log.info("Exception reading external process", e);
156             } finally {
157                 IOUtils.closeQuietly(br);
158             }
159             
160         }
