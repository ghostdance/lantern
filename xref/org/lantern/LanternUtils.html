113     public static void genericWrite(final Object message, 
114         final ChannelFuture future) {
115         final Channel ch = future.getChannel();
116         if (ch.isConnected()) {
117             ch.write(message);
118         } else {
119             future.addListener(new ChannelFutureListener() {
120                 @Override
121                 public void operationComplete(final ChannelFuture cf) 
122                     throws Exception {
123                     if (cf.isSuccess()) {
124                         ch.write(message);
125                     }
126                 }
127             });
128         }
129     }

145     private static Socket openOutgoingPeerSocket(
146         final URI uri, final P2PClient p2pClient,
147         final Map<URI, AtomicInteger> peerFailureCount,
148         final boolean raw) throws IOException {
149 
150         if (p2pClient == null) {
151             LOG.info("P2P client is null. Testing?");
152             throw new IOException("P2P client not connected");
153         }
154         
155         // Start the connection attempt.
156         try {
157             LOG.info("Creating a new socket to {}", uri);
158             final Socket sock;
159             if (raw) {
160                 sock = p2pClient.newRawSocket(uri);
161             } else {
162                 sock = p2pClient.newSocket(uri);
163             }
164             
165             // Note that it's OK that this prints SSL_NULL_WITH_NULL_NULL --
166             // the handshake doesn't actually happen until the first IO, so
167             // the SSL ciphers and such should be all null at this point.
168             LOG.debug("Got outgoing peer socket {}", sock);
169             if (sock instanceof SSLSocket) {
170                 LOG.debug("Socket has ciphers {}", 
171                     ((SSLSocket)sock).getEnabledCipherSuites());
172             } else {
173                 LOG.warn("Not an SSL socket...");
174             }
175             //startReading(sock, browserToProxyChannel, recordStats);
176             return sock;
177         } catch (final NoAnswerException nae) {
178             // This is tricky, as it can mean two things. First, it can mean
179             // the XMPP message was somehow lost. Second, it can also mean
180             // the other side is actually not there and didn't respond as a
181             // result.
182             LOG.info("Did not get answer!! Closing channel from browser", nae);
183             final AtomicInteger count = peerFailureCount.get(uri);
184             if (count == null) {
185                 LOG.info("Incrementing failure count");
186                 peerFailureCount.put(uri, new AtomicInteger(0));
187             }
188             else if (count.incrementAndGet() > 5) {
189                 LOG.info("Got a bunch of failures in a row to this peer. " +
190                     "Removing it.");
191                 
192                 // We still reset it back to zero. Note this all should 
193                 // ideally never happen, and we should be able to use the
194                 // XMPP presence alerts to determine if peers are still valid
195                 // or not.
196                 peerFailureCount.put(uri, new AtomicInteger(0));
197                 //proxyStatusListener.onCouldNotConnectToPeer(uri);
198             } 
199             throw new IOExceptionWithCause(nae);
200         } catch (final IOException ioe) {
201             //proxyStatusListener.onCouldNotConnectToPeer(uri);
202             LOG.warn("Could not connect to peer", ioe);
203             throw ioe;
204         }
205     }

261     private static String macMe(final byte[] mac) {
262         // We wrap the MAC in a SHA-1 to avoid distributing actual 
263         // MAC addresses.
264         final MessageDigest md = new Sha1();
265         md.update(mac);
266         final byte[] raw = md.digest();
267         MAC_ADDRESS = Base64.encodeBase64URLSafeString(raw);
268         return MAC_ADDRESS;
269     }

283     public static boolean isTransferEncodingChunked(final HttpMessage m) {
284         final List<String> chunked = 
285             m.getHeaders(HttpHeaders.Names.TRANSFER_ENCODING);
286         if (chunked.isEmpty()) {
287             return false;
288         }
289 
290         for (String v: chunked) {
291             if (v.equalsIgnoreCase(HttpHeaders.Values.CHUNKED)) {
292                 return true;
293             }
294         }
295         return false;

338     public static Collection<String> toHttpsCandidates(final String uriStr) {
339         final Collection<String> segments = new LinkedHashSet<String>();
340         try {
341             final org.apache.commons.httpclient.URI uri = 
342                 new org.apache.commons.httpclient.URI(uriStr, false);
343             final String host = uri.getHost();
344             //LOG.info("Using host: {}", host);
345             segments.add(host);
346             final String[] segmented = host.split("\\.");
347             //LOG.info("Testing segments: {}", Arrays.asList(segmented));
348             for (int i = 0; i < segmented.length; i++) {
349                 final String tmp = segmented[i];
350                 segmented[i] = "*";
351                 final String segment = StringUtils.join(segmented, '.');
352                 //LOG.info("Adding segment: {}", segment);
353                 segments.add(segment);
354                 segmented[i] = tmp;
355             }
356             
357             for (int i = 1; i < segmented.length - 1; i++) {
358                 final String segment = 
359                     "*." + StringUtils.join(segmented, '.', i, segmented.length);//segmented.slice(i,segmented.length).join(".");
360                 //LOG.info("Checking segment: {}", segment);
361                 segments.add(segment);
362             }
363         } catch (final URIException e) {
364             LOG.error("Could not create URI?", e);
365         }
366         return segments;
367     }

391     public static int randomPort() {
392         if (LanternConstants.ON_APP_ENGINE) {
393             // Can't create server sockets on app engine.
394             return -1;
395         }
396         for (int i = 0; i < 20; i++) {
397             // The +1 on the random int is because 
398             // Math.abs(Integer.MIN_VALUE) == Integer.MIN_VALUE -- caught
399             // by FindBugs.
400             final int randomPort = 1024 + (Math.abs(secureRandom.nextInt() + 1) % 60000);
401             ServerSocket sock = null;
402             try {
403                 sock = new ServerSocket();
404                 sock.bind(new InetSocketAddress("127.0.0.1", randomPort));
405                 final int port = sock.getLocalPort();
406                 return port;
407             } catch (final IOException e) {
408                 LOG.info("Could not bind to port: {}", randomPort);
409             } finally {
410                 if (sock != null) {
411                     try {
412                         sock.close();
413                     } catch (IOException e) {
414                     }
415                 }
416             }
417             
418         }
419         
420         // If we can't grab one of our securely chosen random ports, use
421         // whatever port the OS assigns.
422         ServerSocket sock = null;
423         try {
424             sock = new ServerSocket();
425             sock.bind(null);
426             final int port = sock.getLocalPort();
427             return port;
428         } catch (final IOException e) {
429             LOG.info("Still could not bind?");
430             return 1024 + (Math.abs(secureRandom.nextInt() + 1) % 60000);
431         } finally {
432             if (sock != null) {
433                 try {
434                     sock.close();
435                 } catch (IOException e) {
436                 }
437             }
438         }
439     }

454     private static String findKeytoolPath() {
455 
456         if (SystemUtils.IS_OS_MAC_OSX) {
457             // try to explicitly select the 1.6 keytool -- 
458             // The user may have 1.5 selected as the default 
459             // javavm (default in os x 10.5.8) 
460             // in this case, the default location below will
461             // point to the 1.5 keytool instead.
462             final File keytool16 = new File(
463                 "/System/Library/Frameworks/JavaVM.framework/Versions/1.6/Commands/keytool");
464             if (keytool16.exists()) {
465                 return keytool16.getAbsolutePath();
466             }
467         } 
468         final File jh = new File(System.getProperty("java.home"), "bin");
469         if (jh.isDirectory()) {
470             final String name;
471             if (SystemUtils.IS_OS_WINDOWS) {
472                 name = "keytool.exe";
473             } else {
474                 name = "keytool";
475             }
476             try {
477                 return new File(jh, name).getCanonicalPath();
478             } catch (final IOException e) {
479                 LOG.warn("Error getting canonical path: " + jh);
480             }
481         } else {
482             LOG.warn("java.home/bin not a directory? "+jh);
483         }
484         
485         final File defaultLocation = new File("/usr/bin/keytool");
486         if (defaultLocation.exists()) {
487             return defaultLocation.getAbsolutePath();
488         }
489         final String networkSetupBin = CommandLine.findExecutable("keytool");
490         if (networkSetupBin != null) {
491             return networkSetupBin;
492         }
493         LOG.error("Could not find keytool?!?!?!?");
494         return null;
495     }

709     public static String getRequestHeaders(final HttpServletRequest request) {
710         final Enumeration<String> headers = request.getHeaderNames();
711         final StringBuilder sb = new StringBuilder();
712         sb.append("\n");
713         sb.append(request.getRequestURL().toString());
714         sb.append("\n");
715         while (headers.hasMoreElements()) {
716             final String headerName = headers.nextElement();
717             sb.append(headerName);
718             sb.append(": ");
719             sb.append(request.getHeader(headerName));
720             sb.append("\n");
721         }
722         return sb.toString();

858      * Accesses the object to set a property on with a nested dot notation as
859      * in object1.object2.
860      * 
861      * Public for testing. Note this is actually not use in favor of
862      * ModelMutables that consolidates all accessible methods.
863      */
864     public static Object getTargetForPath(final Object root, final String path) 
865         throws IllegalAccessException, InvocationTargetException, 
866         NoSuchMethodException {
867         if (!path.contains(".")) {
868             return root;
869         }
870         final String curProp = StringUtils.substringBefore(path, ".");
871         final Object propObject = PropertyUtils.getProperty(root, curProp);
872         final String nextProp = StringUtils.substringAfter(path, ".");
873         if (nextProp.contains(".")) {
874             return getTargetForPath(propObject, nextProp);
875         }
876         return propObject;
877     }
