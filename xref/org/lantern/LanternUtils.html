162     public static void genericWrite(final Object message, 
163         final ChannelFuture future) {
164         final Channel ch = future.getChannel();
165         if (ch.isConnected()) {
166             ch.write(message);
167         } else {
168             future.addListener(new ChannelFutureListener() {
169                 @Override
170                 public void operationComplete(final ChannelFuture cf) 
171                     throws Exception {
172                     if (cf.isSuccess()) {
173                         ch.write(message);
174                     }
175                 }
176             });
177         }

192     public static Socket openOutgoingPeerSocket(
193         final URI uri, final P2PClient p2pClient,
194         final Map<URI, AtomicInteger> peerFailureCount,
195         final boolean raw) throws IOException {
196 
197         if (p2pClient == null) {
198             LOG.info("P2P client is null. Testing?");
199             throw new IOException("P2P client not connected");
200         }
201         
202         // Start the connection attempt.
203         try {
204             LOG.info("Creating a new socket to {}", uri);
205             final Socket sock;
206             if (raw) {
207                 sock = p2pClient.newRawSocket(uri);
208             } else {
209                 sock = p2pClient.newSocket(uri);
210             }
211             LOG.info("Got outgoing peer socket: {}", sock);
212             //startReading(sock, browserToProxyChannel, recordStats);
213             return sock;
214         } catch (final NoAnswerException nae) {
215             // This is tricky, as it can mean two things. First, it can mean
216             // the XMPP message was somehow lost. Second, it can also mean
217             // the other side is actually not there and didn't respond as a
218             // result.
219             LOG.info("Did not get answer!! Closing channel from browser", nae);
220             final AtomicInteger count = peerFailureCount.get(uri);
221             if (count == null) {
222                 LOG.info("Incrementing failure count");
223                 peerFailureCount.put(uri, new AtomicInteger(0));
224             }
225             else if (count.incrementAndGet() > 5) {
226                 LOG.info("Got a bunch of failures in a row to this peer. " +
227                     "Removing it.");
228                 
229                 // We still reset it back to zero. Note this all should 
230                 // ideally never happen, and we should be able to use the
231                 // XMPP presence alerts to determine if peers are still valid
232                 // or not.
233                 peerFailureCount.put(uri, new AtomicInteger(0));
234                 //proxyStatusListener.onCouldNotConnectToPeer(uri);
235             } 
236             throw new IOExceptionWithCause(nae);
237         } catch (final IOException ioe) {
238             //proxyStatusListener.onCouldNotConnectToPeer(uri);
239             LOG.warn("Could not connect to peer", ioe);
240             throw ioe;
241         }
242     }

244     public static void startReading(final Socket sock, final Channel channel, 
245         final boolean recordStats) {
246         final Runnable runner = new Runnable() {
247             @Override
248             public void run() {
249                 final byte[] buffer = new byte[4096];
250                 long count = 0;
251                 int n = 0;
252                 try {
253                     final InputStream is = sock.getInputStream();
254                     while (-1 != (n = is.read(buffer))) {
255                         //LOG.info("Writing response data: {}", new String(buffer, 0, n));
256                         // We need to make a copy of the buffer here because
257                         // the writes are asynchronous, so the bytes can
258                         // otherwise get scrambled.
259                         final ChannelBuffer buf =
260                             ChannelBuffers.copiedBuffer(buffer, 0, n);
261                         channel.write(buf);
262                         if (recordStats) {
263                             StatsTracker tracker = LanternHub.statsTracker();
264                             tracker.addBytesProxied(n, sock);
265                             tracker.addDownBytesViaProxies(n, sock);
266                         }
267                         count += n;
268                         
269                     }
270                     ProxyUtils.closeOnFlush(channel);
271 
272                 } catch (final IOException e) {
273                     LOG.info("Exception relaying peer data back to browser",e);
274                     ProxyUtils.closeOnFlush(channel);
275                     
276                     // The other side probably just closed the connection!!
277                     
278                     //channel.close();
279                     //proxyStatusListener.onError(peerUri);
280                     
281                 }
282             }
283         };
284         final Thread peerReadingThread = 
285             new Thread(runner, "Peer-Data-Reading-Thread");
286         peerReadingThread.setDaemon(true);
287         peerReadingThread.start();
288     }

290     public static String getMacAddress() {
291         if (MAC_ADDRESS != null) {
292             LOG.info("Returning MAC: "+MAC_ADDRESS);
293             return MAC_ADDRESS;
294         }
295         final Enumeration<NetworkInterface> nis;
296         try {
297             nis = NetworkInterface.getNetworkInterfaces();
298         } catch (final SocketException e1) {
299             throw new Error("Could not read network interfaces?");
300         }
301         while (nis.hasMoreElements()) {
302             final NetworkInterface ni = nis.nextElement();
303             try {
304                 if (!ni.isUp()) {
305                     LOG.info("Ignoring interface that's not up: {}", 
306                         ni.getDisplayName());
307                     continue;
308                 }
309                 final byte[] mac = ni.getHardwareAddress();
310                 if (mac != null && mac.length > 0) {
311                     LOG.info("Returning 'normal' MAC address");
312                     return macMe(mac);
313                 }
314             } catch (final SocketException e) {
315                 LOG.warn("Could not get MAC address?");
316             }
317         }
318         try {
319             LOG.warn("Returning custom MAC address");
320             return macMe(InetAddress.getLocalHost().getHostAddress() + 
321                     System.currentTimeMillis());
322         } catch (final UnknownHostException e) {
323             final byte[] bytes = new byte[24];
324             LanternHub.secureRandom().nextBytes(bytes);
325             return macMe(bytes);
326         }
327     }

342     private static String macMe(final byte[] mac) {
343         final MessageDigest md = new Sha1();
344         md.update(mac);
345         final byte[] raw = md.digest();
346         MAC_ADDRESS = Base64.encodeBase64URLSafeString(raw);
347         return MAC_ADDRESS;

363     public static boolean isTransferEncodingChunked(final HttpMessage m) {
364         final List<String> chunked = 
365             m.getHeaders(HttpHeaders.Names.TRANSFER_ENCODING);
366         if (chunked.isEmpty()) {
367             return false;
368         }
369 
370         for (String v: chunked) {
371             if (v.equalsIgnoreCase(HttpHeaders.Values.CHUNKED)) {
372                 return true;
373             }
374         }
375         return false;
376     }

472     public static Collection<String> toHttpsCandidates(final String uriStr) {
473         final Collection<String> segments = new LinkedHashSet<String>();
474         try {
475             final org.apache.commons.httpclient.URI uri = 
476                 new org.apache.commons.httpclient.URI(uriStr, false);
477             final String host = uri.getHost();
478             //LOG.info("Using host: {}", host);
479             segments.add(host);
480             final String[] segmented = host.split("\\.");
481             //LOG.info("Testing segments: {}", Arrays.asList(segmented));
482             for (int i = 0; i < segmented.length; i++) {
483                 final String tmp = segmented[i];
484                 segmented[i] = "*";
485                 final String segment = StringUtils.join(segmented, '.');
486                 //LOG.info("Adding segment: {}", segment);
487                 segments.add(segment);
488                 segmented[i] = tmp;
489             }
490             
491             for (int i = 1; i < segmented.length - 1; i++) {
492                 final String segment = 
493                     "*." + StringUtils.join(segmented, '.', i, segmented.length);//segmented.slice(i,segmented.length).join(".");
494                 //LOG.info("Checking segment: {}", segment);
495                 segments.add(segment);
496             }
497         } catch (final URIException e) {
498             LOG.error("Could not create URI?", e);
499         }
500         return segments;
501     }

532     public static int randomPort() {
533         final SecureRandom sr = LanternHub.secureRandom();
534         for (int i = 0; i < 20; i++) {
535             // The +1 on the random int is because 
536             // Math.abs(Integer.MIN_VALUE) == Integer.MIN_VALUE -- caught
537             // by FindBugs.
538             final int randomPort = 1024 + (Math.abs(sr.nextInt() + 1) % 60000);
539             ServerSocket sock = null;
540             try {
541                 sock = new ServerSocket();
542                 sock.bind(new InetSocketAddress("127.0.0.1", randomPort));
543                 final int port = sock.getLocalPort();
544                 return port;
545             } catch (final IOException e) {
546                 LOG.info("Could not bind to port: {}", randomPort);
547             } finally {
548                 if (sock != null) {
549                     try {
550                         sock.close();
551                     } catch (IOException e) {
552                     }
553                 }
554             }
555             
556         }
557         
558         // If we can't grab one of our securely chosen random ports, use
559         // whatever port the OS assigns.
560         ServerSocket sock = null;
561         try {
562             sock = new ServerSocket();
563             sock.bind(null);
564             final int port = sock.getLocalPort();
565             return port;
566         } catch (final IOException e) {
567             LOG.info("Still could not bind?");
568             return 1024 + (Math.abs(sr.nextInt() + 1) % 60000);
569         } finally {
570             if (sock != null) {
571                 try {
572                     sock.close();
573                 } catch (IOException e) {
574                 }
575             }
576         }
577     }

592     private static String findKeytoolPath() {
593 
594         if (SystemUtils.IS_OS_MAC_OSX) {
595             // try to explicitly select the 1.6 keytool -- 
596             // The user may have 1.5 selected as the default 
597             // javavm (default in os x 10.5.8) 
598             // in this case, the default location below will
599             // point to the 1.5 keytool instead.
600             final File keytool16 = new File("/System/Library/Frameworks/JavaVM.framework/Versions/1.6/Commands/keytool");
601             if (keytool16.exists()) {
602                 return keytool16.getAbsolutePath();
603             }
604         } 
605         final File jh = new File(System.getProperty("java.home"), "bin");
606         if (jh.isDirectory()) {
607             final String name;
608             if (SystemUtils.IS_OS_WINDOWS) {
609                 name = "keytool.exe";
610             } else {
611                 name = "keytool";
612             }
613             try {
614                 return new File(jh, name).getCanonicalPath();
615             } catch (final IOException e) {
616                 LOG.warn("Error getting canonical path: " + jh);
617             }
618         } else {
619             LOG.warn("java.home/bin not a directory? "+jh);
620         }
621         
622         final File defaultLocation = new File("/usr/bin/keytool");
623         if (defaultLocation.exists()) {
624             return defaultLocation.getAbsolutePath();
625         }
626         final String networkSetupBin = CommandLine.findExecutable("keytool");
627         if (networkSetupBin != null) {
628             return networkSetupBin;
629         }
630         LOG.error("Could not find keytool?!?!?!?");
631         return null;
632     }

925     public static String getRequestHeaders(final HttpServletRequest request) {
926         final Enumeration<String> headers = request.getHeaderNames();
927         final StringBuilder sb = new StringBuilder();
928         sb.append("\n");
929         sb.append(request.getRequestURL().toString());
930         sb.append("\n");
931         while (headers.hasMoreElements()) {
932             final String headerName = headers.nextElement();
933             sb.append(headerName);
934             sb.append(": ");
935             sb.append(request.getHeader(headerName));
936             sb.append("\n");
937         }
938         return sb.toString();

943      * returns bean property assocated with a method name of the form 
944      * getXyzW -> xyzW or isXyzW -> xyzW

946      * returns null if no property name is deduced

948     public static String methodNameToProperty(final String methodName) {
949         if (methodName.startsWith("get")) {
950             return _lowerFirst(methodName.substring(3));
951         }
952         else if (methodName.startsWith("is")) {
953             return _lowerFirst(methodName.substring(2));
954         }
955         else {

957         }

960     private static String _lowerFirst(final String s) {
961         if (s == null) {
962             return null;
963         }
964         if (s.length() > 1) {
965             return s.substring(0,1).toLowerCase() + s.substring(1);
966         }
967         else {
968             return s.toLowerCase();
969         }
