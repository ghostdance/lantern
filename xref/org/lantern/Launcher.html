87      public Launcher(final String... args) {
88          this.commandLineArgs = args;

90          //Connection.DEBUG_ENABLED = true;
91          Thread.setDefaultUncaughtExceptionHandler(new UncaughtExceptionHandler() {

94                  handleError(e, false);
95              }

99      public void run() {
100         LOG = LoggerFactory.getLogger(Launcher.class);
101         try {
102             launch(this.commandLineArgs);
103         } catch (final Throwable t) {
104             handleError(t, true);
105         }
106         if (SystemUtils.IS_OS_LINUX) {
107             synchronized (this) {
108                 try {
109                     this.wait();
110                 } catch (final InterruptedException e) {
111                     LOG.info("Interrupted", e);
112                 }
113             }
114         }
115     }

128     private void launch(final String... args) {
129         LOG.info("Starting Lantern...");
130         configureCipherSuites();
131 
132         // first apply any command line settings
133         final Options options = buildOptions();
134 
135         final CommandLineParser parser = new PosixParser();
136         final CommandLine cmd;
137         try {
138             cmd = parser.parse(options, args);
139             if (cmd.getArgs().length > 0) {
140                 throw new UnrecognizedOptionException("Extra arguments were provided");
141             }
142         }
143         catch (final ParseException e) {
144             printHelp(options, e.getMessage()+" args: "+Arrays.asList(args));
145             return;
146         }
147         
148         if (cmd.hasOption(OPTION_HELP)) {
149             printHelp(options, null);
150             return;
151         } else if (cmd.hasOption(OPTION_VERSION)) {
152             printVersion();
153             return;
154         }
155 
156         injector = Guice.createInjector(new LanternModule());
157         model = instance(Model.class);
158         set = model.getSettings();
159         
160         // We parse this one separately because we need this value right away.
161         if (cmd.hasOption(OPTION_DISABLE_UI)) {
162             LOG.info("Disabling UI");
163             set.setUiEnabled(false);
164         }
165         else {
166             set.setUiEnabled(true);
167         }
168         
169         censored = instance(Censored.class);
170         threadPublicIpLookup();
171         
172         LOG.debug("Creating display...");
173         final Display display;
174         if (set.isUiEnabled()) {
175             // We initialize this super early in case there are any errors 
176             // during startup we have to display to the user.
177             Display.setAppName("Lantern");
178             //display = injector.getInstance(Display.class);;
179             display = DisplayWrapper.getDisplay();
180             // Also, We need the system tray to listen for events early on.
181             //LanternHub.systemTray().createTray();
182             
183         }
184         else {
185             display = null;
186         }
187         
188         messageService = instance(MessageService.class);
189         xmpp = instance(DefaultXmppHandler.class);
190         jettyLauncher = instance(JettyLauncher.class);
191         if (set.isUiEnabled()) {
192             browserService = instance(BrowserService.class);
193             systemTray = instance(SystemTray.class);
194         }
195         sslProxy = instance(SslHttpProxyServer.class);
196         localCipherProvider = instance(LocalCipherProvider.class);
197         plainTextAnsererRelayProxy = instance(PlainTestRelayHttpProxyServer.class);
198         modelUtils = instance(ModelUtils.class);
199         localProxy = instance(LanternHttpProxyServer.class);
200         internalState = instance(InternalState.class);
201         
202         if (set.isUiEnabled()) {
203             LOG.debug("Starting system tray..");
204             try {
205                 systemTray.start();
206             } catch (final Exception e) {
207                 LOG.error("Error starting tray?", e);
208             }
209             LOG.debug("Started system tray..");
210         }
211         
212         LOG.debug("Processing command line options...");
213         processCommandLineOptions(cmd);
214         LOG.debug("Processed command line options...");
215         
216         shutdownable(ModelIo.class);
217         
218         jettyLauncher.start();
219         xmpp.start();
220         sslProxy.start(false, false);
221         localProxy.start();
222         plainTextAnsererRelayProxy.start(true, false);
223         
224         gnomeAutoStart();
225         
226         // Use our stored STUN servers if available.
227         final Collection<String> stunServers = set.getStunServers();
228         if (stunServers != null && !stunServers.isEmpty()) {
229             LOG.info("Using stored STUN servers: {}", stunServers);
230             StunServerRepository.setStunServers(toSocketAddresses(stunServers));
231         }
232         if (LanternUtils.hasNetworkConnection()) {
233             LOG.info("Got internet...");
234             launchWithOrWithoutUi();
235         } else {
236             // If we're running on startup, it's quite likely we just haven't
237             // connected to the internet yet. Let's wait for an internet
238             // connection and then start Lantern.
239             if (model.isLaunchd() || !set.isUiEnabled()) {
240                 LOG.info("Waiting for internet connection...");
241                 LanternUtils.waitForInternet();
242                 launchWithOrWithoutUi();
243             }
244             // If setup is complete and we're not running on startup, open
245             // the dashboard.
246             else if (model.isSetupComplete()) {
247                 if (set.isUiEnabled()) {
248                     browserService.openBrowserWhenPortReady();
249                 }
250                 //jettyLauncher.openBrowserWhenReady();
251                 // Wait for an internet connection before starting the XMPP
252                 // connection.
253                 LOG.info("Waiting for internet connection...");
254                 LanternUtils.waitForInternet();
255                 launchWithOrWithoutUi();
256             } else {
257                 // If we haven't configured Lantern and don't have an internet
258                 // connection, the problem is that we can't verify the user's
259                 // user name and password when they try to login, so we just
260                 // let them know we can't start Lantern until they have a 
261                 // connection.
262                 // TODO: i18n
263                 
264                 final String msg = 
265                     "We're sorry, but you cannot configure Lantern without " +
266                     "an active connection to the internet. Please try again " +
267                     "when you have an internet connection.";
268                 messageService.showMessage("No Internet", msg);
269                 System.exit(0);
270                 
271                 // The new-ui wants to allow the user to configure this later?
272                 //launchWithOrWithoutUi();
273             }
274         }
275 
276         
277         // This is necessary to keep the tray/menu item up in the case
278         // where we're not launching a browser.
279         if (display != null) {
280             while (!display.isDisposed ()) {
281                 if (!display.readAndDispatch ()) display.sleep ();
282             }
283         }
284     }

287     /**
288      * We thread this because otherwise looking up our public IP address 
289      * over the network can delay the creation of settings altogether. That's
290      * problematic if the UI is waiting on them, for example.
291      */
292     private void threadPublicIpLookup() {
293         if (LanternConstants.ON_APP_ENGINE) {
294             return;
295         }
296         final Thread thread = new Thread(new Runnable() {
297             @Override
298             public void run() {
299                 // This performs the public IP lookup so by the time we set
300                 // GET versus GIVE mode we already know the IP and don't have
301                 // to wait.
302                 
303                 // We get the address here to set it in Connectivity.
304                 final InetAddress ip = 
305                     new PublicIpAddress().getPublicIpAddress();
306                 if (ip == null) {
307                     LOG.info("No IP -- possibly no internet connection");
308                     return;
309                 }
310                 model.getConnectivity().setIp(ip.getHostAddress());
311                 
312                 // The IP is cached at this point.
313                 
314                 try {
315                     final Country count = censored.country();
316                     model.getLocation().setCountry(count.getCode());
317                 } catch (final IOException e) {
318                     LOG.error("Could not get country", e);
319                 }
320                 // If the mode isn't set in the model, set the default.
321                 if (set.getMode() == null || set.getMode() == Mode.none) {
322                     if (censored.isCensored()) {
323                         set.setMode(Mode.get);
324                     } else {
325                         set.setMode(Mode.give);
326                     }
327                 }
328             }
329             
330         }, "Public-IP-Lookup-Thread");
331         thread.setDaemon(true);
332         thread.start();
333     }

339     private <T> T instance(final Class<T> clazz) {
340         final T inst = injector.getInstance(clazz);
341         if (Shutdownable.class.isAssignableFrom(clazz)) {
342             addShutdownHook((Shutdownable) inst);
343         }
344         return inst;
345     }

347     private void addShutdownHook(final Shutdownable service) {
348         LOG.info("Adding shutdown hook for {}", service);
349         // TODO: Add these all to a single list of things to do on shutdown.
350         final Thread serviceHook = new Thread(new Runnable() {

354             }
355         }, "ShutdownHook-For-Service-"+service.getClass().getSimpleName());
356         Runtime.getRuntime().addShutdownHook(serviceHook);
357     }

359     private void gnomeAutoStart() {
360         // Before setup we should just do the default, which is to run on
361         // startup. The user can configure this differently at any point 
362         // hereafter.
363         if (SystemUtils.IS_OS_LINUX && 
364             !LanternConstants.GNOME_AUTOSTART.isFile()) {
365             final File lanternDesktop;
366             final File candidate1 = 
367                 new File(LanternConstants.GNOME_AUTOSTART.getName());
368             final File candidate2 = 
369                 new File("install/linux", LanternConstants.GNOME_AUTOSTART.getName());
370             if (candidate1.isFile()) {
371                 lanternDesktop = candidate1;
372             } else if (candidate2.isFile()){
373                 lanternDesktop = candidate2;
374             } else {
375                 LOG.error("Could not find lantern.desktop file");
376                 return;
377             }
378             try {
379                 final File parent = LanternConstants.GNOME_AUTOSTART.getParentFile();
380                 if (!parent.isDirectory()) {
381                     if (!parent.mkdirs()) {
382                         LOG.error("Could not make dir for gnome autostart: "+parent);
383                         return;
384                     }
385                 }
386                 FileUtils.copyFileToDirectory(lanternDesktop, parent);
387                 
388                 LOG.info("Copied {} to {}", lanternDesktop, parent);
389             } catch (final IOException e) {
390                 LOG.error("Could not configure gnome autostart", e);
391             }
392         }
393     }

395     private void configureCipherSuites() {
396         Security.addProvider(new BouncyCastleProvider());
397         if (!LanternUtils.isUnlimitedKeyStrength()) {
398             if (!SystemUtils.IS_OS_WINDOWS_VISTA) {
399                 LOG.error("No policy files on non-Vista machine!!");
400             }
401             LOG.info("Reverting to weaker ciphers on Vista");
402             IceConfig.setCipherSuites(new String[] {
403                 //"TLS_DHE_RSA_WITH_AES_128_CBC_SHA"
404                 "TLS_ECDHE_RSA_WITH_RC4_128_SHA"
405             });
406         } else {
407             // Note the following just sets what cipher suite the server 
408             // side selects. DHE is for perfect forward secrecy.
409             
410             // We include 128 because we never have enough permissions to 
411             // copy the unlimited strength policy files on Vista, so we have 
412             // to revert back to 128.
413             IceConfig.setCipherSuites(new String[] {
414                 //"TLS_DHE_RSA_WITH_AES_256_CBC_SHA",
415                 //"TLS_DHE_RSA_WITH_AES_128_CBC_SHA"
416                 //"TLS_RSA_WITH_RC4_128_SHA"
417                 "TLS_ECDHE_RSA_WITH_RC4_128_SHA"
418             });
419         }
420     }

423     private static Collection<InetSocketAddress> toSocketAddresses(
424         final Collection<String> stunServers) {
425         final Collection<InetSocketAddress> isas = 
426             new HashSet<InetSocketAddress>();
427         for (final String server : stunServers) {
428             final String host = StringUtils.substringBefore(server, ":");
429             final String port = StringUtils.substringAfter(server, ":");
430             isas.add(new InetSocketAddress(host, Integer.parseInt(port)));
431         }
432         return isas;
433     }

446     private void loadLocalPasswordFile(final String pwFilename) {
447         //final LocalCipherProvider lcp = localCipherProvider;
448         if (!localCipherProvider.requiresAdditionalUserInput()) {
449             LOG.error("Settings do not require a password to unlock.");
450             System.exit(1);
451         }
452 
453         if (StringUtils.isBlank(pwFilename)) {
454             LOG.error("No filename specified to --{}", OPTION_PASSWORD_FILE);
455             System.exit(1);
456         }
457         final File pwFile = new File(pwFilename);
458         if (!(pwFile.exists() && pwFile.canRead())) {
459             LOG.error("Unable to read password from {}", pwFilename);
460             System.exit(1);
461         }
462 
463         LOG.info("Reading local password from file \"{}\"", pwFilename);
464         try {
465             final String pw = FileUtils.readLines(pwFile, "US-ASCII").get(0);
466             final boolean init = !localCipherProvider.isInitialized();
467             localCipherProvider.feedUserInput(pw.toCharArray(), init);
468         }
469         catch (final IndexOutOfBoundsException e) {
470             LOG.error("Password in file \"{}\" was incorrect", pwFilename);
471             System.exit(1);
472         }
473         catch (final InvalidKeyException e) {
474             LOG.error("Password in file \"{}\" was incorrect", pwFilename);
475             System.exit(1);
476         }
477         catch (final GeneralSecurityException e) {
478             LOG.error("Failed to initialize using password in file \"{}\": {}", pwFilename, e);
479             System.exit(1);
480         }
481         catch (final IOException e) {
482             LOG.error("Failed to initialize using password in file \"{}\": {}", pwFilename, e);
483             System.exit(1);
484         }
485     }

514     /**
515      * The autoconnector tries to auto-connect the first time that it observes 
516      * that the settings have reached the SET state.
517      */
518     private class AutoConnector {
519         
520         private boolean done = false;
521         
522         private AutoConnector() {
523             checkAutoConnect();
524             if (!done) {
525                 Events.register(this);
526             }
527         }
528         
529         private void checkAutoConnect() {
530             LOG.info("Checking auto-connect...");
531             if (done) {
532                 return;
533             }
534             
535             // only test once.
536             done = true;
537             
538             LOG.info("Settings loaded, testing auto-connect behavior");
539             // This won't connect in the case where the user hasn't entered 
540             // their user name and password and the user is running with a UI.
541             // Otherwise, it will connect.
542             if (model.getSettings().isAutoConnect() && //LanternHub.settings().isConnectOnLaunch() &&
543                 (modelUtils.isConfigured() || !set.isUiEnabled())) {
544                 final Runnable runner = new Runnable() {
545                     @Override
546                     public void run() {
547                         try {
548                             xmpp.connect();
549                             if (model.getModal() == Modal.gtalkConnecting) {
550                                 internalState.advanceModal(null);
551                             }
552                         } catch (final IOException e) {
553                             LOG.info("Could not login", e);
554                         } catch (final CredentialException e) {
555                             LOG.info("Bad credentials");
556                         } catch (final NotInClosedBetaException e) {
557                             LOG.warn("Not in closed beta!!");
558                         }
559                     }
560                 };
561                 final Thread t = new Thread(runner, "Auto-Starting-Thread");
562                 t.setDaemon(true);
563                 t.start();
564             } else {
565                 LOG.info("Not auto-logging in with model:\n{}", model);
566             }
567         }
568     }

596     private void configureProductionLogger() {
597         final File logDir = LanternConstants.LOG_DIR;
598         final File logFile = new File(logDir, "java.log");
599         final Properties props = new Properties();
600         try {
601             final String logPath = logFile.getCanonicalPath();
602             props.put("log4j.appender.RollingTextFile.File", logPath);
603             props.put("log4j.rootLogger", "warn, RollingTextFile");
604             props.put("log4j.appender.RollingTextFile",
605                     "org.apache.log4j.RollingFileAppender");
606             props.put("log4j.appender.RollingTextFile.MaxFileSize", "1MB");
607             props.put("log4j.appender.RollingTextFile.MaxBackupIndex", "1");
608             props.put("log4j.appender.RollingTextFile.layout",
609                     "org.apache.log4j.PatternLayout");
610             props.put(
611                     "log4j.appender.RollingTextFile.layout.ConversionPattern",
612                     "%-6r %d{ISO8601} %-5p [%t] %c{2}.%M (%F:%L) - %m%n");
613 
614             // This throws and swallows a FileNotFoundException, but it
615             // doesn't matter. Just weird.
616             PropertyConfigurator.configure(props);
617             System.out.println("Set logger file to: " + logPath);
618             final ExceptionalAppenderCallback callback = 
619                 new ExceptionalAppenderCallback() {
620 
621                     @Override
622                     public boolean addData(final JSONObject json, 
623                         final LoggingEvent le) {
624                         if (!set.isAutoReport()) {
625                             // Don't report anything if the user doesn't have
626                             // it turned on.
627                             return false;
628                         }
629                         json.put("version", LanternConstants.VERSION);
630                         return true;
631                     }
632             };
633             final Appender bugAppender = new ExceptionalAppender(
634                LanternConstants.GET_EXCEPTIONAL_API_KEY, callback);
635             BasicConfigurator.configure(bugAppender);
636         } catch (final IOException e) {
637             System.out.println("Exception setting log4j props with file: "
638                     + logFile);
639             e.printStackTrace();
640         }

643     private void handleError(final Throwable t, final boolean exit) {
644         final String msg = msg(t);
645         LOG.error("Uncaught exception: "+msg, t);
646         if (t instanceof SWTError || msg.contains("SWTError")) {
647             System.out.println(
648                 "To run without a UI, run lantern with the --" + 
649                 OPTION_DISABLE_UI +
650                 " command line argument");
651         } 
652         else if (t instanceof UnsatisfiedLinkError && 
653             msg.contains("Cannot load 32-bit SWT libraries on 64-bit JVM")) {
654             messageService.showMessage("Architecture Error",
655                 "We're sorry, but it appears you're running 32-bit Lantern on a 64-bit JVM.");
656         }
657         else if (!lanternStarted && set.isUiEnabled()) {
658             LOG.info("Showing error to user...");
659             messageService.showMessage("Startup Error",
660                "We're sorry, but there was an error starting Lantern " +
661                "described as '"+msg+"'.");
662         }
663         if (exit) {
664             LOG.info("Exiting Lantern");
665             System.exit(1);
666         }
667     }

669     private static String msg(final Throwable t) {
670         final String msg = t.getMessage();
671         if (msg == null) {
672             return "";
673         } 
674         return msg;

759     private void processCommandLineOptions(final CommandLine cmd) {
760 
761         final String secOpt = OPTION_OAUTH2_CLIENT_SECRETS_FILE;
762         if (cmd.hasOption(secOpt)) {
763             modelUtils.loadOAuth2ClientSecretsFile(
764                 cmd.getOptionValue(secOpt));
765         }
766 
767         final String credOpt = OPTION_OAUTH2_USER_CREDENTIALS_FILE;
768         if (cmd.hasOption(credOpt)) {
769             modelUtils.loadOAuth2UserCredentialsFile(
770                 cmd.getOptionValue(credOpt));
771         }
772         
773         //final Settings set = LanternHub.settings();
774         
775         set.setUseTrustedPeers(parseOptionDefaultTrue(cmd, OPTION_TRUSTED_PEERS));
776         set.setUseAnonymousPeers(parseOptionDefaultTrue(cmd, OPTION_ANON_PEERS));
777         set.setUseLaeProxies(parseOptionDefaultTrue(cmd, OPTION_LAE));
778         set.setUseCentralProxies(parseOptionDefaultTrue(cmd, OPTION_CENTRAL));
779         
780         IceConfig.setTcp(parseOptionDefaultTrue(cmd, OPTION_TCP));
781         IceConfig.setUdp(parseOptionDefaultTrue(cmd, OPTION_UDP));
782         
783         /*
784         if (cmd.hasOption(OPTION_USER)) {
785             set.setUserId(cmd.getOptionValue(OPTION_USER));
786         }
787         if (cmd.hasOption(OPTION_PASS)) {
788             set.(cmd.getOptionValue(OPTION_PASS));
789         }
790         */
791         
792         if (cmd.hasOption(OPTION_REFRESH_TOK)) {
793             set.setRefreshToken(cmd.getOptionValue(OPTION_REFRESH_TOK));
794         }
795         if (cmd.hasOption(OPTION_ACCESS_TOK)) {
796             set.setAccessToken(cmd.getOptionValue(OPTION_ACCESS_TOK));
797         }
798         // option to disable use of keychains in local privacy 
799         if (cmd.hasOption(OPTION_DISABLE_KEYCHAIN)) {
800             LOG.info("Disabling use of system keychains");
801             set.setKeychainEnabled(false);
802         }
803         else {
804             set.setKeychainEnabled(true);
805         }
806         
807         if (cmd.hasOption(OPTION_PASSWORD_FILE)) {
808             loadLocalPasswordFile(cmd.getOptionValue(OPTION_PASSWORD_FILE));
809         }
810 
811         if (cmd.hasOption(OPTION_PUBLIC_API)) {
812             set.setBindToLocalhost(false);
813         }
814         if (cmd.hasOption(OPTION_API_PORT)) {
815             final String apiPortStr =
816                 cmd.getOptionValue(OPTION_API_PORT);
817             LOG.info("Using command-line port: "+ apiPortStr);
818             final int apiPort = Integer.parseInt(apiPortStr);
819             StaticSettings.setApiPort(apiPort);
820         } else {
821             LOG.info("Using random port...");
822         }
823         LOG.info("Running API on port: {}", StaticSettings.getApiPort());
824 
825         if (cmd.hasOption(OPTION_SERVER_PORT)) {
826             final String serverPortStr =
827                 cmd.getOptionValue(OPTION_SERVER_PORT);
828             LOG.info("Using command-line proxy port: "+serverPortStr);
829             final int serverPort = Integer.parseInt(serverPortStr);
830             set.setServerPort(serverPort);
831         } else {
832             LOG.info("Using random give mode proxy port...");
833             set.setServerPort(LanternUtils.randomPort());
834         }
835         LOG.info("Running give mode proxy on port: {}", set.getServerPort());
836         
837         if (cmd.hasOption(OPTION_LAUNCHD)) {
838             LOG.info("Running from launchd or launchd set on command line");
839             model.setLaunchd(true);
840         } else {
841             model.setLaunchd(false);
842         }
843         
844         if (cmd.hasOption(OPTION_GIVE)) {
845             model.getSettings().setMode(Mode.give);
846         } else if (cmd.hasOption(OPTION_GET)) {
847             model.getSettings().setMode(Mode.get);
848         }
849         
850         model.setCache(!LanternUtils.isDevMode());
851         if (cmd.hasOption(OPTION_NO_CACHE)) {
852             model.setCache(false);
853         }
854     }
