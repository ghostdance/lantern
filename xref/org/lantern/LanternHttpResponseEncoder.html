11  /**
12   * HTTP response encoder for Lantern that sends responses back to the browser
13   * and keeps statistics on bytes proxied.
14   */
15  public class LanternHttpResponseEncoder extends ProxyHttpResponseEncoder {
16      
17      private final Logger log = LoggerFactory.getLogger(getClass());
18      private final Stats stats;
19      
20  
21      public LanternHttpResponseEncoder(final Stats stats) {
22          super(true);
23          this.stats = stats; 
24      }
25  
26      @Override
27      protected Object encode(final ChannelHandlerContext ctx, 
28          final Channel channel, final Object msg) throws Exception {
29          final ChannelBuffer cb = 
30              (ChannelBuffer) super.encode(ctx, channel, msg);
31          if (cb == null) {
32              return null;
33          }
34          final int bytes = cb.readableBytes();
35          if (msg instanceof ProxyHttpResponse) {
36              // This is called when we just pass unproxied requests over to
37              // LittleProxy.
38              this.stats.addDirectBytes(bytes);
39          } else {
40              // If it's *not* a ProxyHttpResponse, that means it's something
41              // we didn't simply pass to LittleProxy and instead proxied
42              // ourselves. This could be a straight ChannelBuffer from a p2p
43              // socket, relayed non-HTTP CONNECT data from one of our proxies.
44              // HTTP CONNECT data has to be accounted for differently, as it
45              // bypassed any encoder.
46              
47              // global bytes proxied statistic
48              this.stats.addBytesProxied(bytes, channel);
49          }
50          return cb;
51      }
